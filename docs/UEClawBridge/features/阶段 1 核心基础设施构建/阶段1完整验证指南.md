## 5. 阶段1.4 验证（事务保护）

### 5.1 Undo/Redo功能验证

**步骤**：
```python
from tools.universal_proxy import run_ue_python
import unreal

# 获取当前Actor数量
world = unreal.EditorLevelLibrary.get_editor_world()
actors_before = unreal.GameplayStatics.get_all_actors_of_class(world, unreal.Actor)
count_before = len(actors_before)
print(f"操作前Actor数量: {count_before}")

# 创建10个Actor
result = run_ue_python("""
import unreal
for i in range(10):
    loc = unreal.Vector(i*100, 0, 0)
    unreal.EditorLevelLibrary.spawn_actor_from_class(unreal.StaticMeshActor, loc)
""", undo_description="Create 10 Actors")

print(f"执行结果: {result['success']}")
print(f"事务提交: {result['transaction_committed']}")

# 验证Actor增加
actors_after = unreal.GameplayStatics.get_all_actors_of_class(world, unreal.Actor)
count_after = len(actors_after)
print(f"操作后Actor数量: {count_after}")

# 提示用户手动撤销
print("\n💡 请在UE编辑器中按 Ctrl+Z 撤销")
input("按Enter继续...")

# 验证撤销
actors_undo = unreal.GameplayStatics.get_all_actors_of_class(world, unreal.Actor)
count_undo = len(actors_undo)
print(f"撤销后Actor数量: {count_undo}")
print(f"✓ 撤销成功: {count_undo == count_before}")
```

### 7.1 端到端测试脚本

创建 `run_integration_test.py`：

```python
"""
阶段1完整集成测试
"""

import sys
import asyncio
from pathlib import Path

# 添加路径
plugin_root = Path("D:/MyProject/ArtClaw/subprojects/UEDAgentProj/Plugins/UEClawBridge")
python_path = plugin_root / "Content" / "Python"
sys.path.insert(0, str(python_path))

from mcp_server import get_mcp_server
from tools.universal_proxy import run_ue_python


def test_full_pipeline():
    """测试完整流程"""
    print("\n=== 阶段1完整集成测试 ===\n")
    
    # 1. 检查服务器
    server = get_mcp_server()
    assert server and server.is_running, "✗ 服务器未启动"
    print("✓ WebSocket服务器运行正常")
    
    # 2. 测试安全代码
    result = run_ue_python("""
import unreal
actors = unreal.EditorLevelLibrary.get_selected_level_actors()
print(f"选中 {len(actors)} 个Actor")
""", undo_description="Integration Test")
    
    assert result["success"], f"✗ 执行失败: {result['error']}"
    assert result["blocked_by_guard"] == False, "✗ 不应被阻止"
    assert result["transaction_committed"] == True, "✗ 事务应提交"
    print("✓ 安全代码执行成功")
    
    # 3. 测试危险代码被阻止
    result = run_ue_python("import os\nos.system('test')")
    assert not result["success"], "✗ 危险代码应被阻止"
    assert result["blocked_by_guard"] == True, "✗ 应由预审阻止"
    print("✓ 危险代码被正确阻止")
    
    # 4. 测试Undo功能（需要手动验证）
    print("\n💡 请在编辑器中按 Ctrl+Z 验证撤销功能")
    input("   按Enter继续...")
    print("✓ 集成测试完成")

if __name__ == "__main__":
    print("=" * 60)
    print("阶段1集成测试")
    print("=" * 60)
    
    try:
        test_full_pipeline()
        print("\n✅ 所有集成测试通过!")
    except Exception as e:
        print(f"\n❌ 集成测试失败: {e}")
```
