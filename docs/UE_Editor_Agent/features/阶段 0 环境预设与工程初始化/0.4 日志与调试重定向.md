# 阶段 0.4：日志与调试重定向系统 (Logging System)

**目标**：构建多级日志系统，将Python层的所有输出（AI指令、MCP通信、错误堆栈）实时重定向到UE Output Log，支持按级别过滤和分类查看。

**🚩 里程碑**：
1. Python的`print()`输出实时显示在UE Output Log中
2. 支持DEBUG/INFO/WARNING/ERROR四级日志分类
3. MCP通信日志单独分类，便于调试协议交互
4. 完整的错误堆栈以红色高亮显示
5. 可在Output Log过滤器中单独查看Agent日志

### 2. Python日志重定向实现

#### 2.1 创建日志重定向器 (init_unreal.py)
在 `Plugins/UEEditorAgent/Content/Python/init_unreal.py` 中实现：

```python
import unreal
import sys
import io
import logging
from datetime import datetime

# 配置日志级别映射
LOG_LEVEL_MAP = {
    'DEBUG': unreal.PythonLogLevel.VERBOSE,
    'INFO': unreal.PythonLogLevel.LOG,
    'WARNING': unreal.PythonLogLevel.WARNING,
    'ERROR': unreal.PythonLogLevel.ERROR,
}

class UELogRedirector(io.TextIOBase):
    """将Python标准输出重定向到UE日志系统"""
    
    def __init__(self, log_category='LogUEAgent'):
        super().__init__()
        self.log_category = log_category
        self.buffer = ''
        
    def write(self, message):
        """重写write方法，将消息发送到UE"""
        if not message:
            return
            
        # 处理可能的Unicode和多行消息
        msg_str = str(message)
        
        # 按行分割，避免单条日志过长
        lines = msg_str.split('\n')
        for line in lines:
            line = line.strip()
            if line:  # 跳过空行
                unreal.log_line(LOG_LEVEL_MAP['INFO'], self.log_category, line)
                
    def flush(self):
        """Python标准接口，无需实现"""
        pass

class UEMCPLogRedirector(io.TextIOBase):
    """MCP通信专用日志重定向器"""
    
    def __init__(self):
        super().__init__()
        
    def write(self, message):
        if not message:
            return
            
        msg_str = str(message).strip()
        if msg_str:
            # MCP日志使用单独的类别，便于过滤
            unreal.log_line(unreal.PythonLogLevel.LOG, 'LogUEAgent_MCP', msg_str)
            
    def flush(self):
        pass

class UELogger:
    """提供类似logging模块的接口，支持多级别日志"""
    
    @staticmethod
    def debug(message):
        """调试信息（详细）"""
        unreal.log_line(unreal.PythonLogLevel.VERBOSE, 'LogUEAgent', f"[DEBUG] {message}")
        
    @staticmethod
    def info(message):
        """普通信息"""
        unreal.log_line(unreal.PythonLogLevel.LOG, 'LogUEAgent', f"[INFO] {message}")
        
    @staticmethod
    def warning(message):
        """警告信息"""
        unreal.log_line(unreal.PythonLogLevel.WARNING, 'LogUEAgent_Warning', f"[WARNING] {message}")
        
    @staticmethod
    def error(message):
        """错误信息"""
        unreal.log_line(unreal.PythonLogLevel.ERROR, 'LogUEAgent_Error', f"[ERROR] {message}")
        
    @staticmethod
    def exception(message, exc_info=None):
        """异常堆栈"""
        import traceback
        unreal.log_line(unreal.PythonLogLevel.ERROR, 'LogUEAgent_Error', f"[EXCEPTION] {message}")
        if exc_info:
            tb_lines = traceback.format_exception(*exc_info)
            for line in tb_lines:
                unreal.log_line(unreal.PythonLogLevel.ERROR, 'LogUEAgent_Error', line.strip())

def initialize_logging():
    """初始化日志系统"""
    # 保存原始stdout/stderr
    global _original_stdout, _original_stderr
    _original_stdout = sys.stdout
    _original_stderr = sys.stderr
    
    # 创建重定向器
    sys.stdout = UELogRedirector('LogUEAgent')
    sys.stderr = UELogRedirector('LogUEAgent_Error')
    
    # 设置Python内置logging
    logging.basicConfig(
        level=logging.DEBUG,
        format='[%(levelname)s] %(message)s'
    )
    
    # 记录初始化完成
    UELogger.info("UE Agent logging system initialized")
    UELogger.debug("Debug mode enabled")

# 自动初始化
initialize_logging()
```

### 3. MCP通信日志拦截

#### 3.1 创建MCP日志装饰器
```python
import functools
import json

class MCPSessionLogger:
    """自动记录MCP通信"""
    
    def __init__(self):
        self.session_id = datetime.now().strftime('%Y%m%d_%H%M%S')
        
    def log_request(self, method, params):
        """记录请求"""
        UELogger.info(f"[MCP Request] {method}")
        UELogger.debug(f"  Params: {json.dumps(params, indent=2)}")
        
    def log_response(self, result):
        """记录响应"""
        UELogger.info(f"[MCP Response] Success")
        UELogger.debug(f"  Result: {json.dumps(result, indent=2)}")
        
    def log_error(self, error):
        """记录错误"""
        UELogger.error(f"[MCP Error] {error}")

# 全局MCP日志实例
mcp_logger = MCPSessionLogger()

def log_mcp_call(func):
    """装饰器：自动记录MCP调用"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        method = func.__name__
        mcp_logger.log_request(method, kwargs)
        
        try:
            result = func(*args, **kwargs)
            mcp_logger.log_response(result)
            return result
        except Exception as e:
            mcp_logger.log_error(str(e))
            raise
    
    return wrapper

# 使用示例
@log_mcp_call
def run_ue_python(code: str):
    """执行Python代码的MCP工具"""
    # 实际执行逻辑
    UELogger.info(f"Executing Python code: {code[:50]}...")
    # ...
    return {"success": True}
```

### 4. 错误堆栈美化

#### 4.1 创建错误处理模块
```python
import traceback
import sys

class ErrorHandler:
    """统一错误处理和日志记录"""
    
    @staticmethod
    def handle_exception(exc_type, exc_value, exc_traceback):
        """全局异常处理"""
        # 跳过KeyboardInterrupt
        if issubclass(exc_type, KeyboardInterrupt):
            sys.__excepthook__(exc_type, exc_value, exc_traceback)
            return
        
        # 记录异常信息
        UELogger.error("=" * 60)
        UELogger.error("UNHANDLED EXCEPTION")
        UELogger.error("=" * 60)
        UELogger.error(f"Type: {exc_type.__name__}")
        UELogger.error(f"Message: {str(exc_value)}")
        UELogger.error("-" * 60)
        UELogger.error("Traceback:")
        
        # 格式化堆栈
        tb_list = traceback.format_exception(exc_type, exc_value, exc_traceback)
        for line in ''.join(tb_list).split('\n'):
            if line.strip():
                UELogger.error(line)
        
        UELogger.error("=" * 60)

def install_exception_handler():
    """安装全局异常处理器"""
    sys.excepthook = ErrorHandler.handle_exception

# 自动安装异常处理器
install_exception_handler()
```

def set_log_level(level: str):
    """动态设置日志级别"""
    valid_levels = ['DEBUG', 'INFO', 'WARNING', 'ERROR']
    if level.upper() in valid_levels:
        global CURRENT_LOG_LEVEL
        CURRENT_LOG_LEVEL = level.upper()
        UELogger.info(f"Log level changed to {CURRENT_LOG_LEVEL}")
    else:
        UELogger.error(f"Invalid log level: {level}")
