# 阶段 0.5：依赖隔离与自动安装 (Dependency Management)

**目标**：在插件启动时自动检测Python环境，按需安装`websockets`、`pydantic`等必要库到插件私有目录，避免污染引擎自带的Python环境。

**🚩 里程碑**：
1. 插件启动时自动检测`site-packages`目录是否存在必要依赖
2. 缺失依赖时自动执行`pip install --target`安装到插件目录
3. 安装过程静默执行，失败时优雅降级并提供手动安装指引
4. 支持离线环境，提供预打包的依赖包
5. 所有依赖安装在`Plugins/UEEditorAgent/Content/Python/Lib`目录下

### 2. Python依赖检测与安装模块

#### 2.1 创建依赖管理器 (dependency_manager.py)
在 `Plugins/UEEditorAgent/Content/Python/` 目录下创建：

```python
import unreal
import sys
import os
import subprocess
import json
import importlib.util
from pathlib import Path
from typing import List, Dict, Tuple

try:
    from init_unreal import UELogger
except ImportError:
    # 降级方案
    class UELogger:
        @staticmethod
        def info(msg): print(f"[INFO] {msg}")
        @staticmethod
        def warning(msg): print(f"[WARNING] {msg}")
        @staticmethod
        def error(msg): print(f"[ERROR] {msg}")

class DependencyManager:
    """管理Python依赖的检测和自动安装"""
    
    def __init__(self):
        # 获取插件根目录
        self.plugin_root = Path(unreal.Paths.project_plugins_dir()) / "UEEditorAgent"
        self.python_root = self.plugin_root / "Content" / "Python"
        self.lib_dir = self.python_root / "Lib"
        self.config_file = self.python_root / "dependency_config.json"
        self.requirements_file = self.python_root / "requirements.txt"
        
        # 确保目录存在
        self.lib_dir.mkdir(parents=True, exist_ok=True)
        
        # 添加到Python路径
        if str(self.lib_dir) not in sys.path:
            sys.path.insert(0, str(self.lib_dir))
    
    def load_config(self) -> Dict:
        """加载依赖配置"""
        try:
            if self.config_file.exists():
                with open(self.config_file, 'r', encoding='utf-8') as f:
                    return json.load(f)
        except Exception as e:
            UELogger.error(f"Failed to load config: {e}")
        
        # 默认配置
        return {
            "dependencies": [
                {"name": "websockets", "min_version": "12.0", "required": True},
                {"name": "pydantic", "min_version": "2.0.0", "required": True}
            ],
            "install_dir": "Lib",
            "timeout_seconds": 120
        }
    
    def check_package(self, package_name: str) -> bool:
        """检查包是否已安装"""
        try:
            spec = importlib.util.find_spec(package_name)
            return spec is not None
        except:
            return False
    
    def check_package_version(self, package_name: str, min_version: str) -> bool:
        """检查包版本是否满足要求"""
        try:
            if not self.check_package(package_name):
                return False
                
            module = importlib.import_module(package_name)
            if hasattr(module, '__version__'):
                installed_version = module.__version__
                return self._version_compare(installed_version, min_version) >= 0
            return True  # 无版本信息，假设满足
        except Exception as e:
            UELogger.warning(f"Failed to check version for {package_name}: {e}")
            return False
    
    def _version_compare(self, v1: str, v2: str) -> int:
        """比较版本号"""
        # 简单版本比较
        def parse_version(v):
            return [int(x) for x in v.split('.') if x.isdigit()]
        
        parts1 = parse_version(v1)
        parts2 = parse_version(v2)
        
        for i in range(max(len(parts1), len(parts2))):
            p1 = parts1[i] if i < len(parts1) else 0
            p2 = parts2[i] if i < len(parts2) else 0
            if p1 > p2:
                return 1
            elif p1 < p2:
                return -1
        return 0
    
    def get_missing_packages(self) -> List[Tuple[str, str, bool]]:
        """获取缺失的包列表"""
        config = self.load_config()
        missing = []
        
        for dep in config.get("dependencies", []):
            name = dep.get("name")
            min_version = dep.get("min_version", "0.0.0")
            required = dep.get("required", True)
            description = dep.get("description", "")
            
            if not self.check_package_version(name, min_version):
                missing.append((name, min_version, required, description))
        
        return missing
    
    def install_package(self, package_name: str, min_version: str, timeout: int = 120) -> bool:
        """安装单个包"""
        try:
            UELogger.info(f"Installing {package_name}>={min_version} to {self.lib_dir}")
            
            # 构造pip命令
            cmd = [
                sys.executable,  # Python解释器路径
                "-m",
                "pip",
                "install",
                f"{package_name}>={min_version}",
                "--target", str(self.lib_dir),
                "--no-warn-script-location",  # 避免警告
                "--disable-pip-version-check",  # 禁用版本检查
            ]
            
            # 添加清华镜像源（中国大陆加速）
            cmd.extend([
                "-i", "https://pypi.tuna.tsinghua.edu.cn/simple",
                "--trusted-host", "pypi.tuna.tsinghua.edu.cn"
            ])
            
            UELogger.info(f"Running: {' '.join(cmd)}")
            
            # 执行安装
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=timeout,
                check=False  # 不抛出异常，手动检查返回码
            )
            
            if result.returncode == 0:
                UELogger.info(f"Successfully installed {package_name}")
                if result.stdout:
                    UELogger.debug(f"Install output: {result.stdout}")
                return True
            else:
                UELogger.error(f"Failed to install {package_name}")
                if result.stderr:
                    UELogger.error(f"Error: {result.stderr}")
                return False
                
        except subprocess.TimeoutExpired:
            UELogger.error(f"Installation timeout for {package_name}")
            return False
        except Exception as e:
            UELogger.error(f"Installation error: {e}")
            return False
    
    def install_missing_packages(self) -> bool:
        """安装所有缺失的包"""
        missing = self.get_missing_packages()
        
        if not missing:
            UELogger.info("All dependencies are satisfied ✓")
            return True
        
        UELogger.info(f"Found {len(missing)} missing packages")
        
        success_count = 0
        for name, version, required, desc in missing:
            if required:
                UELogger.warning(f"Required package missing: {name} ({desc})")
                if self.install_package(name, version):
                    success_count += 1
                else:
                    UELogger.error(f"Failed to install required package: {name}")
                    return False  # 必需包安装失败
            else:
                UELogger.info(f"Optional package: {name} ({desc})")
                self.install_package(name, version)
                # 可选包失败不阻止启动
        
        UELogger.info(f"Installed {success_count}/{len(missing)} packages")
        return True
    
    def verify_installation(self) -> bool:
        """验证安装是否成功"""
        missing = self.get_missing_packages()
        
        required_missing = [
            (name, desc) for name, _, required, desc in missing if required
        ]
        
        if required_missing:
            UELogger.error("Required packages still missing after installation:")
            for name, desc in required_missing:
                UELogger.error(f"  - {name}: {desc}")
            return False
        
        UELogger.info("All required packages verified ✓")
        return True
    
    def create_offline_bundle(self):
        """创建离线安装包（用于分发）"""
        import zipfile
        
        bundle_file = self.plugin_root / f"PythonDeps_{datetime.now().strftime('%Y%m%d')}.zip"
        
        UELogger.info(f"Creating offline bundle: {bundle_file}")
        
        with zipfile.ZipFile(bundle_file, 'w', zipfile.ZIP_DEFLATED) as zipf:
            # 添加依赖目录
            for root, dirs, files in os.walk(self.lib_dir):
                for file in files:
                    file_path = Path(root) / file
                    arc_name = file_path.relative_to(self.plugin_root)
                    zipf.write(file_path, arc_name)
        
        UELogger.info(f"Bundle created: {bundle_file}")
        return bundle_file
    
    def extract_offline_bundle(self, bundle_path: str) -> bool:
        """解压离线安装包"""
        import zipfile
        
        bundle_file = Path(bundle_path)
        if not bundle_file.exists():
            UELogger.error(f"Bundle file not found: {bundle_path}")
            return False
        
        try:
            UELogger.info(f"Extracting bundle: {bundle_file}")
            
            with zipfile.ZipFile(bundle_file, 'r') as zipf:
                # 解压到插件目录
                zipf.extractall(self.plugin_root)
            
            UELogger.info("Bundle extracted successfully")
            return True
        except Exception as e:
            UELogger.error(f"Failed to extract bundle: {e}")
            return False
```

void FUEEditorAgentModule::RunDependencyCheck()
{
    UE_LOG(LogUEAgent, Log, TEXT("Checking Python dependencies..."));
    
    // 在Python中执行依赖检查
    FString PythonCommand = R"(
import sys
sys.path.append('")" + FPaths::ProjectPluginsDir() + TEXT("/UEEditorAgent/Content/Python") + R"("')

try:
    from dependency_manager import DependencyManager
    dm = DependencyManager()
    
    # 检查并安装依赖
    if dm.install_missing_packages():
        if dm.verify_installation():
            print("✓ All dependencies installed successfully")
        else:
            print("✗ Verification failed")
            sys.exit(1)
    else:
        print("✗ Installation failed")
        sys.exit(1)
        
except Exception as e:
    print(f"Error during dependency check: {e}")
    import traceback
    traceback.print_exc()
    sys.exit(1)
)"";
    
    FPythonCommandEx PythonEx;
    PythonEx.Command = PythonCommand;
    PythonEx.ExecutionMode = EPythonExecutionMode::ExecuteFile;
    PythonEx.FileExecutionScope = EPythonFileExecutionScope::Public;
    
    IPluginManager& PluginManager = IPluginManager::Get();
    TSharedPtr<IPlugin> Plugin = PluginManager.FindPlugin(TEXT("UEEditorAgent"));
    if (Plugin.IsValid())
    {
        FString PluginBaseDir = Plugin->GetBaseDir();
        PythonEx.FileExecutionScope = EPythonFileExecutionScope::Private;
    }
    
    // 执行Python代码
    FPythonScriptPlugin::Get()->ExecPythonCommandEx(PythonEx);
    
    UE_LOG(LogUEAgent, Log, TEXT("Dependency check completed"));
}

### 4. 离线安装支持

#### 4.1 创建离线安装脚本 (install_deps_offline.py)
```python
"""
离线安装脚本 - 用于没有网络的环境
用法：python install_deps_offline.py <bundle_file.zip>
"""

import sys
import argparse
from pathlib import Path

# 添加插件路径
plugin_root = Path(__file__).parent.parent
sys.path.insert(0, str(plugin_root / "Content" / "Python"))

from dependency_manager import DependencyManager

def main():
    parser = argparse.ArgumentParser(description='Install UE Agent dependencies offline')
    parser.add_argument('bundle', help='Path to the dependency bundle zip file')
    args = parser.parse_args()
    
    dm = DependencyManager()
    
    print(f"Extracting bundle: {args.bundle}")
    if dm.extract_offline_bundle(args.bundle):
        print("✓ Offline installation completed")
        
        if dm.verify_installation():
            print("✓ All packages verified")
            return 0
        else:
            print("✗ Package verification failed")
            return 1
    else:
        print("✗ Installation failed")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```

### 7. 常见问题

#### 7.1 pip命令找不到
**问题**：`python -m pip`执行失败
**解决**：
- 确保UE的Python环境包含pip
- 手动在引擎Python目录执行`python -m ensurepip`

#### 7.2 权限错误
**问题**：无法写入Lib目录
**解决**：
- 以管理员身份运行UE编辑器
- 检查插件目录的写入权限

#### 7.3 版本冲突
**问题**：与引擎自带包版本冲突
**解决**：
- 使用`--target`确保隔离安装
- 在`sys.path`中优先加载Lib目录
