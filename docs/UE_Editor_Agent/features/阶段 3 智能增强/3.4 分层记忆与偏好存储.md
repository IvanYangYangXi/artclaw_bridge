## 🔧 核心实现

```python
import json
import os
from typing import Dict, Any, List
import time

class MemoryStore:
    """记忆存储"""
    
    def __init__(self, storage_path: str):
        self.storage_path = storage_path
        self.short_term: List[Dict[str, Any]] = []  # 短期记忆（最近10条）
        self.long_term: Dict[str, Any] = {}  # 长期记忆
        self.preferences: Dict[str, Any] = {}  # 用户偏好
        self.max_short_term = 10
        
        self._load()
    
    def add_short_term(self, event: str, data: Any):
        """添加短期记忆"""
        self.short_term.append({
            "timestamp": time.time(),
            "event": event,
            "data": data
        })
        
        # 保持最近N条
        if len(self.short_term) > self.max_short_term:
            self.short_term = self.short_term[-self.max_short_term:]
    
    def add_long_term(self, key: str, value: Any):
        """添加长期记忆"""
        self.long_term[key] = {
            "value": value,
            "timestamp": time.time()
        }
        self._save()
    
    def set_preference(self, key: str, value: Any):
        """设置用户偏好"""
        self.preferences[key] = value
        self._save()
    
    def get_preference(self, key: str, default: Any = None) -> Any:
        """获取用户偏好"""
        return self.preferences.get(key, default)
    
    def get_recent(self, event_type: str, limit: int = 5) -> List[Any]:
        """获取最近记忆"""
        recent = []
        for memory in reversed(self.short_term):
            if memory["event"] == event_type:
                recent.append(memory["data"])
                if len(recent) >= limit:
                    break
        return recent
    
    def _load(self):
        """从磁盘加载"""
        if os.path.exists(self.storage_path):
            try:
                with open(self.storage_path, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    self.long_term = data.get("long_term", {})
                    self.preferences = data.get("preferences", {})
            except Exception as e:
                print(f"[Memory] 加载失败: {e}")
    
    def _save(self):
        """保存到磁盘"""
        try:
            os.makedirs(os.path.dirname(self.storage_path), exist_ok=True)
            with open(self.storage_path, 'w', encoding='utf-8') as f:
                json.dump({
                    "long_term": self.long_term,
                    "preferences": self.preferences
                }, f, indent=2, ensure_ascii=False)
        except Exception as e:
            print(f"[Memory] 保存失败: {e}")


# 使用示例
if __name__ == "__main__":
    memory = MemoryStore("./data/memory.json")
    
    # 添加记忆
    memory.add_short_term("create_actor", {"name": "MyActor"})
    memory.add_long_term("favorite_actor_class", "StaticMeshActor")
    memory.set_preference("auto_save", True)
    
    # 检索
    print("最近创建的Actor:", memory.get_recent("create_actor"))
    print("偏好设置:", memory.get_preference("auto_save"))
