### 核心目标
构建**接口影子化层**，在调用UE API时自动检测版本差异，提供兼容性支持，确保插件在不同UE版本间兼容。

import unreal
from typing import Any, Callable
import functools

class VersionCompatibilityLayer:
    """版本兼容层"""
    
    def __init__(self):
        self.version = unreal.SystemLibrary.get_engine_version()
        self.major_version = int(self.version.split('.')[0])
        self.minor_version = int(self.version.split('.')[1])
        
        self.compatibility_map = {
            ("spawn_actor", 5, 0): self._spawn_actor_legacy,
            ("spawn_actor", 5, 3): self._spawn_actor_new,
        }
    
    def call(self, api_name: str, *args, **kwargs) -> Any:
        """调用API并自动处理版本兼容"""
        key = (api_name, self.major_version, self.minor_version)
        
        if key in self.compatibility_map:
            # 使用兼容版本
            method = self.compatibility_map[key]
            return method(*args, **kwargs)
        else:
            # 默认使用最新版本
            return self._call_latest(api_name, *args, **kwargs)
    
    def _call_latest(self, api_name: str, *args, **kwargs):
        """调用最新版本API"""
        if hasattr(unreal.EditorLevelLibrary, api_name):
            method = getattr(unreal.EditorLevelLibrary, api_name)
            return method(*args, **kwargs)
        else:
            raise AttributeError(f"API未找到: {api_name}")
    
    def _spawn_actor_legacy(self, actor_class, location):
        """UE 5.0兼容版本"""
        return unreal.EditorLevelLibrary.spawn_actor_from_class(
            actor_class, location, unreal.Rotator(0, 0, 0)
        )
    
    def _spawn_actor_new(self, actor_class, location, rotation=None):
        """UE 5.3+新版本"""
        if rotation is None:
            rotation = unreal.Rotator(0, 0, 0)
        return unreal.EditorLevelLibrary.spawn_actor_from_class(
            actor_class, location, rotation
        )


# 装饰器
def compatible_api(api_name: str):
    """API兼容装饰器"""
    def decorator(func: Callable):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            compat_layer = VersionCompatibilityLayer()
            return compat_layer.call(api_name, *args, **kwargs)
        return wrapper
    return decorator


# 使用示例
@compatible_api("spawn_actor")
def spawn_actor_safe(actor_class, location):
    """安全生成Actor（自动处理版本兼容）"""
    pass


if __name__ == "__main__":
    compat = VersionCompatibilityLayer()
    print(f"UE版本: {compat.version}")
