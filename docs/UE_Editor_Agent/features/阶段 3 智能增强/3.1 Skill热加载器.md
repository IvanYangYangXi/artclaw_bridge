## 🎯 目标与里程碑

### 核心目标
构建**Skill热加载器**，无需重启编辑器即可加载/更新/卸载AI技能（Skill），实现插件功能的动态扩展。UE插件作为**Skill管理中心**，向所有MCP客户端（WorkBuddy、OpenClaw等）提供统一的Skill访问接口。

### 验收标准
- [ ] Skill加载时间<2秒
- [ ] 热更新不中断当前会话
- [ ] 支持版本回滚（保留最近3个版本）
- [ ] 加载失败不影响系统稳定性
- [ ] MCP接口`list_skills`可访问
- [ ] MCP接口`execute_skill`可执行
- [ ] Skill重载后自动通知所有连接的MCP客户端
- [ ] WorkBuddy/OpenClaw可实时感知Skill列表变化
- [ ] 向MCP客户端暴露Skill列表和执行接口
- [ ] 客户端无需重启即可使用新加载的Skill

## 🔧 核心实现

```python
import os
import json
import importlib
import sys
from typing import Dict, Any, List, Optional
import time
import shutil

class SkillLoader:
    """Skill加载器"""
    
    def __init__(self, skills_dir: str):
        self.skills_dir = skills_dir
        self.loaded_skills: Dict[str, Any] = {}
        self.skill_versions: Dict[str, List[str]] = {}  # 版本历史
    
    def load_skill(self, skill_name: str) -> bool:
        """加载Skill"""
        skill_path = os.path.join(self.skills_dir, skill_name, "main.py")
        
        if not os.path.exists(skill_path):
            print(f"[SkillLoader] Skill未找到: {skill_name}")
            return False
        
        try:
            # 动态导入
            spec = importlib.util.spec_from_file_location(skill_name, skill_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            # 保存Skill
            self.loaded_skills[skill_name] = module
            
            # 保存版本
            if skill_name not in self.skill_versions:
                self.skill_versions[skill_name] = []
            self.skill_versions[skill_name].append(time.strftime("%Y%m%d_%H%M%S"))
            
            print(f"[SkillLoader] Skill加载成功: {skill_name}")
            return True
            
        except Exception as e:
            print(f"[SkillLoader] Skill加载失败: {skill_name} - {e}")
            return False
    
    def reload_skill(self, skill_name: str) -> bool:
        """热重载Skill"""
        if skill_name in self.loaded_skills:
            # 卸载旧版本
            del self.loaded_skills[skill_name]
        
        # 清除模块缓存
        if skill_name in sys.modules:
            del sys.modules[skill_name]
        
        # 加载新版本
        return self.load_skill(skill_name)
    
    def unload_skill(self, skill_name: str) -> bool:
        """卸载Skill"""
        if skill_name in self.loaded_skills:
            del self.loaded_skills[skill_name]
            print(f"[SkillLoader] Skill卸载: {skill_name}")
            return True
        return False
    
    def get_loaded_skills(self) -> List[str]:
        """获取已加载的Skill列表"""
        return list(self.loaded_skills.keys())


# MCP接口暴露（向WorkBuddy/OpenClaw提供访问）

class McpSkillInterface:
    """MCP Skill接口 - 向所有客户端暴露Skill管理功能"""
    
    def __init__(self, skill_loader: SkillLoader):
        self.loader = skill_loader
        self.connected_clients = []  # 连接的MCP客户端
    
    def register_with_mcp(self, mcp_server):
        """注册MCP工具"""
        
        @mcp_server.tool()
        async def list_skills() -> Dict[str, Any]:
            """列出所有已加载的Skill（供WorkBuddy/OpenClaw调用）"""
            skills = self.loader.get_loaded_skills()
            return {
                "skills": [
                    {
                        "name": name,
                        "version": self.loader.skill_versions.get(name, ["unknown"])[-1]
                    }
                    for name in skills
                ]
            }
        
        @mcp_server.tool()
        async def execute_skill(skill_name: str, parameters: Dict) -> Dict[str, Any]:
            """执行指定的Skill"""
            if skill_name not in self.loader.loaded_skills:
                return {"error": f"Skill未找到: {skill_name}"}
            
            try:
                skill = self.loader.loaded_skills[skill_name]
                result = await skill.execute(parameters)
                return {"success": True, "result": result}
            except Exception as e:
                return {"error": str(e)}
        
        @mcp_server.tool()
        async def install_skill(package_path: str) -> Dict[str, Any]:
            """安装Skill包到UE插件"""
            try:
                # 解压并安装Skill
                success = self.loader.install_from_package(package_path)
                if success:
                    # 通知所有客户端
                    await self.notify_clients_skill_installed()
                return {"success": success}
            except Exception as e:
                return {"error": str(e)}
        
        @mcp_server.tool()
        async def reload_skill(skill_name: str) -> Dict[str, Any]:
            """热重载指定的Skill"""
            success = self.loader.reload_skill(skill_name)
            if success:
                await self.notify_clients_skill_reloaded(skill_name)
            return {"success": success}
    
    def add_client(self, client):
        """添加MCP客户端连接"""
        self.connected_clients.append(client)
    
    async def notify_clients_skill_reloaded(self, skill_name: str):
        """通知所有客户端Skill已重载"""
        for client in self.connected_clients:
            await client.send_notification({
                "method": "notifications/skills/reloaded",
                "params": {
                    "skill_name": skill_name,
                    "timestamp": time.time()
                }
            })
    
    async def notify_clients_skill_installed(self):
        """通知所有客户端Skill列表已更新"""
        for client in self.connected_clients:
            await client.send_notification({
                "method": "notifications/skills/updated",
                "params": {
                    "skills": self.loader.get_loaded_skills()
                }
            })


# 完整使用示例
class UEAgentSkillManager:
    """UE Agent Skill管理器 - 统一管理所有Skill"""
    
    def __init__(self, skills_dir: str = "./skills"):
        self.skills_dir = skills_dir
        self.loader = SkillLoader(skills_dir)
        self.mcp_interface = McpSkillInterface(self.loader)
        self.setup_file_watcher()
    
    def setup_file_watcher(self):
        """设置文件监控，实现热加载"""
        try:
            from watchdog.observers import Observer
            from watchdog.events import FileSystemEventHandler
            
            class SkillFileHandler(FileSystemEventHandler):
                def __init__(self, manager):
                    self.manager = manager
                
                def on_modified(self, event):
                    if event.src_path.endswith('.py'):
                        skill_name = os.path.basename(os.path.dirname(event.src_path))
                        unreal.log(f"[UEAgent] Skill文件变化: {skill_name}")
                        
                        # 热重载Skill
                        success = self.manager.loader.reload_skill(skill_name)
                        if success:
                            # 异步通知所有客户端
                            asyncio.create_task(
                                self.manager.mcp_interface.notify_clients_skill_reloaded(skill_name)
                            )
            
            observer = Observer()
            handler = SkillFileHandler(self)
            observer.schedule(handler, self.skills_dir, recursive=True)
            observer.start()
            unreal.log("[UEAgent] Skill文件监控已启动")
            
        except ImportError:
            unreal.log_warning("[UEAgent] watchdog未安装，Skill热加载不可用")
    
    def initialize(self, mcp_server):
        """初始化 - 注册MCP接口"""
        # 加载所有Skill
        self.load_all_skills()
        
        # 注册MCP接口
        self.mcp_interface.register_with_mcp(mcp_server)
        
        unreal.log(f"[UEAgent] Skill管理器初始化完成，已加载 {len(self.loader.get_loaded_skills())} 个Skill")
    
    def load_all_skills(self):
        """加载所有Skill"""
        if not os.path.exists(self.skills_dir):
            os.makedirs(self.skills_dir)
            return
        
        for skill_name in os.listdir(self.skills_dir):
            skill_path = os.path.join(self.skills_dir, skill_name)
            if os.path.isdir(skill_path):
                self.loader.load_skill(skill_name)

## 🤝 与MCP客户端协作流程

### 场景1：WorkBuddy使用UE插件的Skill

```
1. 用户输入: "帮我优化选中的材质"
   ↓
2. WorkBuddy分析意图 → 需要调用材质优化Skill
   ↓
3. WorkBuddy通过MCP调用: unreal://skills/execute
   {
     "skill_name": "batch_material_optimizer",
     "parameters": {"target_actors": [...]}
   }
   ↓
4. UE插件接收调用 → 执行Skill → 返回结果
   ↓
5. WorkBuddy展示结果给用户
```

### 场景2：Skill热重载通知所有客户端

```
1. 开发者修改Skill文件: batch_material_optimizer.py
   ↓
2. UE插件文件监控检测到变化
   ↓
3. UE插件热重载Skill
   ↓
4. UE插件通过MCP推送通知给所有客户端:
   {
     "method": "notifications/skills/reloaded",
     "params": {"skill_name": "batch_material_optimizer"}
   }
   ↓
5. WorkBuddy接收通知 → 更新Skill列表缓存
   ↓
6. OpenClaw接收通知 → 更新Skill列表缓存
```

### 场景3：从远程市场安装Skill

```
1. 用户在WorkBuddy浏览Skill市场
   ↓
2. 用户选择安装: batch-material-optimizer-1.0.clawpkg
   ↓
3. WorkBuddy下载Skill包 → 发送到UE插件:
   unreal://skills/install
   {"package_path": ".../batch-material-optimizer-1.0.clawpkg"}
   ↓
4. UE插件解压并安装Skill → 热加载
   ↓
5. UE插件通知所有客户端: Skill列表已更新
   ↓
6. WorkBuddy和OpenClaw同步更新Skill列表
```
