## 🎯 目标与里程碑

### 核心目标
构建**Skill热加载器**，无需重启编辑器即可加载/更新/卸载AI技能（Skill），实现插件功能的动态扩展。

### 验收标准
- [ ] Skill加载时间<2秒
- [ ] 热更新不中断当前会话
- [ ] 支持版本回滚（保留最近3个版本）
- [ ] 加载失败不影响系统稳定性

## 🔧 核心实现

```python
import os
import json
import importlib
import sys
from typing import Dict, Any, List, Optional
import time
import shutil

class SkillLoader:
    """Skill加载器"""
    
    def __init__(self, skills_dir: str):
        self.skills_dir = skills_dir
        self.loaded_skills: Dict[str, Any] = {}
        self.skill_versions: Dict[str, List[str]] = {}  # 版本历史
    
    def load_skill(self, skill_name: str) -> bool:
        """加载Skill"""
        skill_path = os.path.join(self.skills_dir, skill_name, "main.py")
        
        if not os.path.exists(skill_path):
            print(f"[SkillLoader] Skill未找到: {skill_name}")
            return False
        
        try:
            # 动态导入
            spec = importlib.util.spec_from_file_location(skill_name, skill_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            # 保存Skill
            self.loaded_skills[skill_name] = module
            
            # 保存版本
            if skill_name not in self.skill_versions:
                self.skill_versions[skill_name] = []
            self.skill_versions[skill_name].append(time.strftime("%Y%m%d_%H%M%S"))
            
            print(f"[SkillLoader] Skill加载成功: {skill_name}")
            return True
            
        except Exception as e:
            print(f"[SkillLoader] Skill加载失败: {skill_name} - {e}")
            return False
    
    def reload_skill(self, skill_name: str) -> bool:
        """热重载Skill"""
        if skill_name in self.loaded_skills:
            # 卸载旧版本
            del self.loaded_skills[skill_name]
        
        # 清除模块缓存
        if skill_name in sys.modules:
            del sys.modules[skill_name]
        
        # 加载新版本
        return self.load_skill(skill_name)
    
    def unload_skill(self, skill_name: str) -> bool:
        """卸载Skill"""
        if skill_name in self.loaded_skills:
            del self.loaded_skills[skill_name]
            print(f"[SkillLoader] Skill卸载: {skill_name}")
            return True
        return False
    
    def get_loaded_skills(self) -> List[str]:
        """获取已加载的Skill列表"""
        return list(self.loaded_skills.keys())


# 使用示例
if __name__ == "__main__":
    loader = SkillLoader("./skills")
    loader.load_skill("spawn_tools")
    print(f"已加载Skills: {loader.get_loaded_skills()}")
