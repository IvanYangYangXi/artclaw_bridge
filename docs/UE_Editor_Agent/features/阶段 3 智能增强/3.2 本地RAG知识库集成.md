import os
import chardet
from typing import List, Dict, Any
import numpy as np

class SimpleVectorStore:
    """简化的向量存储"""
    
    def __init__(self, dimension: int = 384):
        self.dimension = dimension
        self.documents: List[Dict[str, Any]] = []
        self.vectors: List[np.ndarray] = []
    
    def add_document(self, text: str, metadata: Dict[str, Any]):
        """添加文档"""
        # 简化的向量化（实际应使用sentence-transformers）
        vector = self._simple_vectorize(text)
        
        self.documents.append({
            "text": text,
            "metadata": metadata
        })
        self.vectors.append(vector)
    
    def search(self, query: str, top_k: int = 5) -> List[Dict[str, Any]]:
        """搜索相关文档"""
        query_vector = self._simple_vectorize(query)
        
        scores = []
        for i, doc_vector in enumerate(self.vectors):
            # 余弦相似度
            similarity = np.dot(query_vector, doc_vector) / (
                np.linalg.norm(query_vector) * np.linalg.norm(doc_vector)
            )
            scores.append((i, similarity))
        
        # 排序
        scores.sort(key=lambda x: x[1], reverse=True)
        
        results = []
        for i, score in scores[:top_k]:
            results.append({
                **self.documents[i],
                "score": float(score)
            })
        
        return results
    
    def _simple_vectorize(self, text: str) -> np.ndarray:
        """简化的向量化（实际应使用Embedding模型）"""
        # 实际项目中应使用：from sentence_transformers import SentenceTransformer
        # model = SentenceTransformer('all-MiniLM-L6-v2')
        # return model.encode(text)
        
        # 简化的哈希向量化
        import hashlib
        vector = np.zeros(self.dimension)
        
        for i, word in enumerate(text.split()[:100]):  # 限制单词数
            hash_val = int(hashlib.md5(word.encode()).hexdigest(), 16)
            idx = hash_val % self.dimension
            vector[idx] += 1
        
        return vector / (np.linalg.norm(vector) + 1e-8)


class RAGKnowledgeBase:
    """RAG知识库"""
    
    def __init__(self, project_dir: str):
        self.project_dir = project_dir
        self.vector_store = SimpleVectorStore()
        self.supported_extensions = {".md", ".txt", ".py", ".cpp", ".h", ".cs"}
    
    def build_index(self):
        """构建向量索引"""
        print(f"[RAG] 开始构建知识库索引: {self.project_dir}")
        
        for root, _, files in os.walk(self.project_dir):
            for file in files:
                if any(file.endswith(ext) for ext in self.supported_extensions):
                    file_path = os.path.join(root, file)
                    self._index_file(file_path)
        
        print(f"[RAG] 索引构建完成，共{len(self.vector_store.documents)}个文档")
    
    def _index_file(self, file_path: str):
        """索引单个文件"""
        try:
            # 检测编码
            with open(file_path, 'rb') as f:
                raw_data = f.read()
                encoding = chardet.detect(raw_data)['encoding'] or 'utf-8'
            
            with open(file_path, 'r', encoding=encoding, errors='ignore') as f:
                content = f.read()
            
            # 分段索引（避免单个文档过大）
            chunk_size = 500
            chunks = [content[i:i+chunk_size] for i in range(0, len(content), chunk_size)]
            
            for i, chunk in enumerate(chunks):
                self.vector_store.add_document(
                    text=chunk,
                    metadata={
                        "file_path": file_path,
                        "chunk_index": i,
                        "total_chunks": len(chunks)
                    }
                )
                
        except Exception as e:
            print(f"[RAG] 索引文件失败: {file_path} - {e}")
    
    def query(self, question: str, top_k: int = 5) -> List[Dict[str, Any]]:
        """查询知识库"""
        print(f"[RAG] 查询: {question}")
        results = self.vector_store.search(question, top_k)
        
        for result in results:
            print(f"[RAG] 检索到: {result['metadata']['file_path']} (分数: {result['score']:.2f})")
        
        return results
