## 🎯 目标与里程碑

### 核心目标
构建**语义检索系统**，根据用户查询自动检索相关文档并**动态注入到Prompt**，增强AI回答的项目相关性。

### 验收标准
- [ ] 检索结果相关性>80%
- [ ] Prompt注入不影响格式
- [ ] Token使用效率提升>30%
- [ ] 支持上下文压缩

## 🔧 核心实现

```python
from typing import List, Dict, Any

class PromptEnhancer:
    """Prompt增强器"""
    
    def __init__(self, rag_kb):
        self.rag_kb = rag_kb
        self.max_context_length = 2000  # Token限制
    
    def enhance_prompt(self, user_query: str, base_prompt: str) -> str:
        """增强Prompt"""
        # 检索相关知识
        relevant_docs = self.rag_kb.query(user_query, top_k=3)
        
        if not relevant_docs:
            return base_prompt
        
        # 构建上下文
        context = self._build_context(relevant_docs)
        
        # 注入到Prompt
        enhanced_prompt = f"""{base_prompt}

# 项目相关知识（自动检索）
以下信息来自项目文档，请参考：

{context}

# 用户查询
{user_query}
"""
        
        return enhanced_prompt
    
    def _build_context(self, docs: List[Dict[str, Any]]) -> str:
        """构建上下文"""
        context_parts = []
        
        for i, doc in enumerate(docs):
            file_path = doc["metadata"]["file_path"]
            text = doc["text"]
            score = doc["score"]
            
            # 压缩文本以适应Token限制
            compressed = self._compress_text(text, max_chars=500)
            
            context_parts.append(f"""
## 参考文档 {i+1} (相关度: {score:.2f})
文件: {file_path}
内容: {compressed}
""")
        
        return "\n".join(context_parts)
    
    def _compress_text(self, text: str, max_chars: int) -> str:
        """压缩文本"""
        if len(text) <= max_chars:
            return text
        
        # 取开头和结尾
        half = max_chars // 2
        return text[:half] + "\n...\n" + text[-half:]


# 使用示例
if __name__ == "__main__":
    from rag_knowledge_base import RAGKnowledgeBase
    
    kb = RAGKnowledgeBase("./docs")
    kb.build_index()
    
    enhancer = PromptEnhancer(kb)
    
    base_prompt = "你是一个UE助手，请回答用户问题。"
    user_query = "如何创建材质?"
    
    enhanced = enhancer.enhance_prompt(user_query, base_prompt)
    print(enhanced)
