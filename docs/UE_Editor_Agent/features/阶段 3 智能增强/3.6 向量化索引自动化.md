import os
import time
from typing import Set
import threading

class IndexAutomator:
    """索引自动化器"""
    
    def __init__(self, rag_kb):
        self.rag_kb = rag_kb
        self.watched_dir = rag_kb.project_dir
        self.file_snapshots: Dict[str, float] = {}
        self.is_running = False
    
    def start_watching(self):
        """开始监控文件变化"""
        self.is_running = True
        self._take_snapshot()
        
        def watch_loop():
            while self.is_running:
                changes = self._detect_changes()
                if changes:
                    print(f"[IndexAuto] 检测到{len(changes)}个文件变化")
                    self._update_index(changes)
                    self._take_snapshot()
                
                time.sleep(5)  # 每5秒检查一次
        
        threading.Thread(target=watch_loop, daemon=True).start()
    
    def stop_watching(self):
        """停止监控"""
        self.is_running = False
    
    def _take_snapshot(self):
        """拍摄文件快照"""
        self.file_snapshots = {}
        
        for root, _, files in os.walk(self.watched_dir):
            for file in files:
                if file.endswith(('.md', '.txt', '.py')):
                    file_path = os.path.join(root, file)
                    try:
                        mtime = os.path.getmtime(file_path)
                        self.file_snapshots[file_path] = mtime
                    except:
                        pass
    
    def _detect_changes(self) -> Set[str]:
        """检测文件变化"""
        changes = set()
        current_files = set()
        
        for root, _, files in os.walk(self.watched_dir):
            for file in files:
                if file.endswith(('.md', '.txt', '.py')):
                    file_path = os.path.join(root, file)
                    current_files.add(file_path)
                    
                    try:
                        mtime = os.path.getmtime(file_path)
                        
                        if file_path not in self.file_snapshots:
                            # 新增文件
                            changes.add(file_path)
                        elif mtime > self.file_snapshots[file_path]:
                            # 修改文件
                            changes.add(file_path)
                    except:
                        pass
        
        # 检测删除的文件
        deleted_files = set(self.file_snapshots.keys()) - current_files
        changes.update(deleted_files)
        
        return changes
    
    def _update_index(self, changes: Set[str]):
        """更新索引"""
        print(f"[IndexAuto] 更新索引: {len(changes)}个文件")
        
        # 简化的更新：直接重建
        # 实际应增量更新
        self.rag_kb.build_index()


# 使用示例
if __name__ == "__main__":
    from rag_knowledge_base import RAGKnowledgeBase
    
    kb = RAGKnowledgeBase("./docs")
    kb.build_index()
    
    automator = IndexAutomator(kb)
    automator.start_watching()
    
    print("开始监控文件变化...")
    time.sleep(60)  # 监控1分钟
    
    automator.stop_watching()
