## 🎯 目标与里程碑

### 核心目标
构建**主线程调度器**，确保所有UE编辑器操作都在游戏主线程执行，解决**线程安全问题**。

### 验收标准
- [ ] 100% UE API调用在主线程执行
- [ ] 异步操作回调正确处理
- [ ] 性能开销<5ms（单次调度）
- [ ] 支持超时控制（默认30秒）
- [ ] 错误捕获和日志完整

---

## 📚 背景与上下文

### 为什么需要GameThread Marshalling？

UE编辑器API**只能在主线程调用**，如果在其他线程调用会导致：

```python
# ❌ 错误示例：在Python线程中直接调用UE API
import threading
import unreal

def bad_idea():
    # 这是在Python线程，不是UE主线程！
    actor = unreal.EditorLevelLibrary.spawn_actor_from_class(
        unreal.StaticMeshActor, unreal.Vector(0, 0, 0)
    )  # 💥 崩溃或断言失败！

thread = threading.Thread(target=bad_idea)
thread.start()
```

典型错误：
- `Assertion failed: IsInGameThread()`
- 随机崩溃
- 数据竞争
- 编辑器无响应

### 设计原则

1. **透明化**：调用方无需关心线程问题
2. **异步支持**：支持回调和Future模式
3. **超时控制**：防止死锁
4. **错误传播**：异常正确传递回调用方

## 🔧 详细实现步骤

### 步骤1：核心调度器实现

创建`GameThread/game_thread_scheduler.py`：

```python
import unreal
import threading
import time
from typing import Callable, Any, Optional, TypeVar, Generic
from concurrent.futures import Future, TimeoutError
import queue
import weakref

T = TypeVar('T')

class GameThreadTask:
    """主线程任务包装器"""
    
    def __init__(self, func: Callable[[], T], name: str = ""):
        self.func = func
        self.name = name or func.__name__
        self.future: Future[T] = Future()
        self.submit_time = time.time()
        self.start_time: Optional[float] = None
        self.end_time: Optional[float] = None
    
    @property
    def duration(self) -> Optional[float]:
        """执行耗时"""
        if self.start_time and self.end_time:
            return self.end_time - self.start_time
        return None
    
    @property
    def wait_time(self) -> float:
        """等待调度时间"""
        if self.start_time:
            return self.start_time - self.submit_time
        return time.time() - self.submit_time


class GameThreadScheduler:
    """
    主线程调度器
    将任务安全地调度到UE主线程执行
    """
    
    def __init__(self):
        self.task_queue: queue.Queue[GameThreadTask] = queue.Queue()
        self.is_running = False
        self.worker_thread: Optional[threading.Thread] = None
        self.tick_handle: Optional[unreal.FDelegateHandle] = None
        
        # 统计
        self.stats = {
            "total_tasks": 0,
            "completed_tasks": 0,
            "failed_tasks": 0,
            "timeout_tasks": 0,
            "avg_wait_time": 0.0,
            "avg_duration": 0.0
        }
        
        # 配置
        self.config = {
            "default_timeout": 30.0,  # 默认超时（秒）
            "max_queue_size": 1000,   # 最大队列长度
            "tick_interval": 0.01,    # 轮询间隔（秒）
            "enable_stats": True
        }
    
    def start(self):
        """启动调度器"""
        if self.is_running:
            return
        
        self.is_running = True
        
        # 注册Tick函数
        self.tick_handle = unreal.register_slate_pre_tick_callback(self._on_tick)
        
        unreal.log("[GameThreadScheduler] 主线程调度器已启动")
    
    def stop(self):
        """停止调度器"""
        if not self.is_running:
            return
        
        self.is_running = False
        
        # 注销Tick回调
        if self.tick_handle:
            unreal.unregister_slate_pre_tick_callback(self.tick_handle)
            self.tick_handle = None
        
        unreal.log("[GameThreadScheduler] 主线程调度器已停止")
    
    def _on_tick(self, delta_time: float):
        """Slate Tick回调，运行在主线程"""
        if not self.is_running:
            return
        
        # 处理队列中的任务
        while not self.task_queue.empty():
            try:
                task = self.task_queue.get_nowait()
                self._execute_task(task)
            except queue.Empty:
                break
    
    def _execute_task(self, task: GameThreadTask):
        """执行任务"""
        try:
            task.start_time = time.time()
            
            # 检查是否已超时
            if task.future.cancelled():
                self.stats["timeout_tasks"] += 1
                unreal.log_warning(f"[GameThread] 任务超时取消: {task.name}")
                return
            
            # 执行任务
            result = task.func()
            
            task.end_time = time.time()
            task.future.set_result(result)
            
            self.stats["completed_tasks"] += 1
            
            # 更新统计
            if self.config["enable_stats"]:
                self._update_stats(task)
            
            unreal.log_verbose(f"[GameThread] 任务完成: {task.name} ({task.duration:.3f}s)")
            
        except Exception as e:
            task.end_time = time.time()
            task.future.set_exception(e)
            
            self.stats["failed_tasks"] += 1
            
            unreal.log_error(f"[GameThread] 任务失败: {task.name} - {e}")
    
    def _update_stats(self, task: GameThreadTask):
        """更新统计信息"""
        n = self.stats["completed_tasks"]
        if n > 1:
            # 移动平均
            self.stats["avg_wait_time"] = (
                (self.stats["avg_wait_time"] * (n - 1) + task.wait_time) / n
            )
            if task.duration:
                self.stats["avg_duration"] = (
                    (self.stats["avg_duration"] * (n - 1) + task.duration) / n
                )
        else:
            self.stats["avg_wait_time"] = task.wait_time
            self.stats["avg_duration"] = task.duration or 0.0
    
    # ========================================================================
    # 任务提交接口
    # ========================================================================
    
    def run_on_game_thread(self, func: Callable[[], T], 
                          name: str = "",
                          timeout: Optional[float] = None) -> T:
        """
        在主线程同步执行任务
        
        Args:
            func: 要执行的函数
            name: 任务名称
            timeout: 超时时间（秒）
            
        Returns:
            函数返回值
            
        Raises:
            TimeoutError: 任务超时
            Exception: 任务执行异常
        """
        # 如果已经在主线程，直接执行
        if self._is_game_thread():
            return func()
        
        task = GameThreadTask(func, name)
        
        # 检查队列长度
        if self.task_queue.qsize() >= self.config["max_queue_size"]:
            raise RuntimeError(f"任务队列已满({self.config['max_queue_size']})")
        
        self.task_queue.put(task)
        self.stats["total_tasks"] += 1
        
        # 等待结果
        timeout = timeout or self.config["default_timeout"]
        try:
            return task.future.result(timeout=timeout)
        except TimeoutError:
            task.future.cancel()
            self.stats["timeout_tasks"] += 1
            unreal.log_error(f"[GameThread] 任务超时: {task.name}")
            raise
    
    def run_async(self, func: Callable[[], T], 
                 name: str = "",
                 callback: Optional[Callable[[Future[T]], None]] = None) -> Future[T]:
        """
        在主线程异步执行任务
        
        Args:
            func: 要执行的函数
            name: 任务名称
            callback: 完成回调
            
        Returns:
            Future对象
        """
        task = GameThreadTask(func, name)
        
        if self.task_queue.qsize() >= self.config["max_queue_size"]:
            raise RuntimeError(f"任务队列已满({self.config['max_queue_size']})")
        
        self.task_queue.put(task)
        self.stats["total_tasks"] += 1
        
        # 添加回调
        if callback:
            task.future.add_done_callback(callback)
        
        return task.future
    
    def _is_game_thread(self) -> bool:
        """检查当前是否在主线程"""
        # 注意：这个检查在Python中不可靠，实际依赖UE的IsInGameThread()
        # 这里只是近似检查
        return threading.current_thread().name == "MainThread"
    
    # ========================================================================
    # 快捷方法
    # ========================================================================
    
    def spawn_actor(self, actor_class: type, location: unreal.Vector, 
                   rotation: unreal.Rotator = unreal.Rotator(0, 0, 0),
                   name: str = "") -> unreal.Actor:
        """在主线程生成Actor"""
        def _spawn():
            return unreal.EditorLevelLibrary.spawn_actor_from_class(
                actor_class, location, rotation
            )
        
        return self.run_on_game_thread(_spawn, name or f"生成{actor_class.__name__}")
    
    def delete_actor(self, actor: unreal.Actor, name: str = "") -> bool:
        """在主线程删除Actor"""
        def _delete():
            return actor.destroy_actor()
        
        return self.run_on_game_thread(_delete, name or f"删除{actor.get_name()}")
    
    def rename_actor(self, actor: unreal.Actor, new_name: str) -> bool:
        """在主线程重命名Actor"""
        def _rename():
            actor.set_actor_label(new_name)
            return True
        
        return self.run_on_game_thread(_rename, f"重命名{actor.get_name()}")
    
    def load_asset(self, asset_path: str) -> unreal.Object:
        """在主线程加载资产"""
        def _load():
            return unreal.EditorAssetLibrary.load_asset(asset_path)
        
        return self.run_on_game_thread(_load, f"加载资产{asset_path}")
    
    def save_asset(self, asset: unreal.Object) -> bool:
        """在主线程保存资产"""
        def _save():
            return unreal.EditorAssetLibrary.save_asset(asset.get_path_name())
        
        return self.run_on_game_thread(_save, f"保存资产{asset.get_name()}")
    
    # ========================================================================
    # 统计和监控
    # ========================================================================
    
    def get_stats(self) -> Dict[str, Any]:
        """获取统计信息"""
        return self.stats.copy()
    
    def get_queue_status(self) -> Dict[str, Any]:
        """获取队列状态"""
        return {
            "queue_size": self.task_queue.qsize(),
            "is_running": self.is_running,
            "current_depth": len(self.transaction_stack) if hasattr(self, 'transaction_stack') else 0
        }
    
    def reset_stats(self):
        """重置统计"""
        self.stats = {
            "total_tasks": 0,
            "completed_tasks": 0,
            "failed_tasks": 0,
            "timeout_tasks": 0,
            "avg_wait_time": 0.0,
            "avg_duration": 0.0
        }


# 全局实例
_scheduler_instance: Optional[GameThreadScheduler] = None

def get_game_thread_scheduler() -> GameThreadScheduler:
    """获取全局调度器实例"""
    global _scheduler_instance
    if _scheduler_instance is None:
        _scheduler_instance = GameThreadScheduler()
        _scheduler_instance.start()
    return _scheduler_instance


# 装饰器支持
def run_on_game_thread(name: str = ""):
    """
    装饰器：确保函数在主线程执行
    
    Usage:
        @run_on_game_thread("创建材质")
        def create_material():
            return unreal.AssetTools.create_asset(...)
    """
    def decorator(func: Callable):
        def wrapper(*args, **kwargs):
            scheduler = get_game_thread_scheduler()
            
            # 如果已经在主线程，直接执行
            if scheduler._is_game_thread():
                return func(*args, **kwargs)
            
            # 否则调度到主线程
            def _call():
                return func(*args, **kwargs)
            
            return scheduler.run_on_game_thread(_call, name or func.__name__)
        
        return wrapper
    
    return decorator


# 使用示例
if __name__ == "__main__":
    scheduler = get_game_thread_scheduler()
    
    # 示例1：同步执行
    try:
        location = unreal.Vector(0, 0, 0)
        actor = scheduler.spawn_actor(unreal.StaticMeshActor, location, name="测试Actor")
        unreal.log(f"成功生成Actor: {actor.get_name()}")
    except Exception as e:
        unreal.log_error(f"生成Actor失败: {e}")
    
    # 示例2：异步执行
    def on_complete(future: Future):
        try:
            actor = future.result()
            unreal.log(f"异步生成完成: {actor.get_name()}")
        except Exception as e:
            unreal.log_error(f"异步生成失败: {e}")
    
    future = scheduler.run_async(
        lambda: scheduler.spawn_actor(unreal.StaticMeshActor, unreal.Vector(100, 0, 0)),
        name="异步生成",
        callback=on_complete
    )
    
    # 示例3：批量操作
    batch = scheduler.batch("批量生成")
    for i in range(10):
        location = unreal.Vector(i * 100, 0, 0)
        batch.add(lambda loc=location: scheduler.spawn_actor(unreal.StaticMeshActor, loc))
    
    try:
        results = batch.execute()
        unreal.log(f"批量生成成功: {len(results)} 个Actor")
    except Exception as e:
        unreal.log_error(f"批量生成失败: {e}")
```

## ❓ 常见问题

### Q1: 任务长时间不执行怎么办？
A:
1. 检查调度器是否启动：`scheduler.is_running`
2. 检查队列长度：`scheduler.get_queue_status()`
3. 确认在主循环中：`unreal.register_slate_pre_tick_callback`

### Q2: 如何取消已提交的任务？
A: Future模式支持取消：
```python
future = scheduler.run_async(...)
future.cancel()  # 取消任务
```

### Q3: 内存泄漏？
A: 确保调用`scheduler.stop()`释放资源，或使用弱引用：
```python
import weakref
scheduler_ref = weakref.ref(scheduler)
```
