import requests
from typing import Dict, Any, List

class CommunityClient:
    """社区客户端"""
    
    def __init__(self, marketplace_url: str = "https://market.openclaw.io"):
        self.marketplace_url = marketplace_url
        self.api_base = f"{marketplace_url}/api/v1"
    
    def browse_skills(self, category: str = None, search: str = None) -> List[Dict[str, Any]]:
        """浏览技能"""
        params = {}
        if category:
            params["category"] = category
        if search:
            params["search"] = search
        
        try:
            response = requests.get(
                f"{self.api_base}/skills",
                params=params,
                timeout=10
            )
            response.raise_for_status()
            
            return response.json().get("skills", [])
            
        except Exception as e:
            print(f"[Community] 浏览失败: {e}")
            return []
    
    def download_skill(self, skill_id: str, download_dir: str) -> str:
        """下载技能"""
        try:
            response = requests.get(
                f"{self.api_base}/skills/{skill_id}/download",
                timeout=30
            )
            response.raise_for_status()
            
            # 保存文件
            file_path = os.path.join(download_dir, f"{skill_id}.openclaw")
            os.makedirs(download_dir, exist_ok=True)
            
            with open(file_path, 'wb') as f:
                f.write(response.content)
            
            print(f"[Community] 技能下载成功: {skill_id}")
            return file_path
            
        except Exception as e:
            print(f"[Community] 下载失败: {e}")
            return None
    
    def rate_skill(self, skill_id: str, rating: int, comment: str = "") -> bool:
        """评价技能"""
        try:
            response = requests.post(
                f"{self.api_base}/skills/{skill_id}/ratings",
                json={
                    "rating": rating,
                    "comment": comment
                },
                timeout=10
            )
            response.raise_for_status()
            
            print(f"[Community] 评价成功: {skill_id}")
            return True
            
        except Exception as e:
            print(f"[Community] 评价失败: {e}")
            return False
    
    def check_updates(self, installed_skills: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """检查更新"""
        try:
            skill_ids = [skill["id"] for skill in installed_skills]
            
            response = requests.post(
                f"{self.api_base}/skills/updates",
                json={"skills": skill_ids},
                timeout=10
            )
            response.raise_for_status()
            
            return response.json().get("updates", [])
            
        except Exception as e:
            print(f"[Community] 更新检查失败: {e}")
            return []


# 使用示例
