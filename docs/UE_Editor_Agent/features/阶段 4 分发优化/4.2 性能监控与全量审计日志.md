import json
import time
import psutil
import os
from datetime import datetime
from typing import Dict, Any, Optional
import threading

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self, log_dir: str):
        self.log_dir = log_dir
        self.session_id = f"session_{int(time.time())}"
        os.makedirs(log_dir, exist_ok=True)
        
        # 当前操作上下文
        self.current_operation: Optional[Dict[str, Any]] = None
        self.start_time: Optional[float] = None
        self.start_memory: Optional[float] = None
    
    def start_operation(self, operation_type: str, details: Dict[str, Any]):
        """开始记录操作"""
        self.current_operation = {
            "session_id": self.session_id,
            "operation_type": operation_type,
            "details": details,
            "timestamp": datetime.now().isoformat()
        }
        self.start_time = time.time()
        self.start_memory = psutil.Process().memory_info().rss / 1024 / 1024  # MB
        
        self.log_event("operation_start", self.current_operation)
    
    def end_operation(self, success: bool, result: Any = None, error: str = None):
        """结束记录操作"""
        if not self.current_operation or not self.start_time:
            return
        
        end_time = time.time()
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        
        duration = end_time - self.start_time
        memory_used = end_memory - (self.start_memory or 0)
        
        log_entry = {
            **self.current_operation,
            "duration_seconds": duration,
            "memory_used_mb": memory_used,
            "success": success,
            "result": str(result) if result else None,
            "error": error
        }
        
        self.log_event("operation_end", log_entry)
        
        # 重置
        self.current_operation = None
        self.start_time = None
        self.start_memory = None
    
    def log_event(self, event_type: str, data: Dict[str, Any]):
        """记录事件"""
        log_file = self._get_log_file()
        
        event = {
            "event_type": event_type,
            "timestamp": datetime.now().isoformat(),
            "data": data
        }
        
        with open(log_file, 'a', encoding='utf-8') as f:
            f.write(json.dumps(event, ensure_ascii=False) + '\n')
    
    def _get_log_file(self) -> str:
        """获取当前日志文件"""
        today = datetime.now().strftime('%Y%m%d')
        return os.path.join(self.log_dir, f"audit_{today}.log")
    
    def get_system_metrics(self) -> Dict[str, Any]:
        """获取系统指标"""
        process = psutil.Process()
        
        return {
            "cpu_percent": psutil.cpu_percent(),
            "memory_mb": process.memory_info().rss / 1024 / 1024,
            "threads": process.num_threads(),
            "handles": process.num_handles() if hasattr(process, 'num_handles') else 0
        }


class AuditLogger:
    """审计日志器"""
    
    def __init__(self, log_dir: str):
        self.monitor = PerformanceMonitor(log_dir)
        self.enabled = True
    
    def log_request(self, request: Dict[str, Any]):
        """记录AI请求"""
        if not self.enabled:
            return
        
        self.monitor.log_event("ai_request", {
            "session_id": self.monitor.session_id,
            "request": request,
            "metrics": self.monitor.get_system_metrics()
        })
    
    def log_response(self, response: Dict[str, Any], latency_ms: float):
        """记录AI响应"""
        if not self.enabled:
            return
        
        self.monitor.log_event("ai_response", {
            "session_id": self.monitor.session_id,
            "response": response,
            "latency_ms": latency_ms,
            "metrics": self.monitor.get_system_metrics()
        })
    
    def log_tool_execution(self, tool_name: str, parameters: Dict[str, Any]):
        """记录工具执行"""
        self.monitor.start_operation("tool_execution", {
            "tool_name": tool_name,
            "parameters": parameters
        })
    
    def end_tool_execution(self, success: bool, result: Any = None, error: str = None):
        """结束工具执行记录"""
        self.monitor.end_operation(success, result, error)


# 全局实例
audit_logger: Optional[AuditLogger] = None

def get_audit_logger() -> AuditLogger:
    """获取全局审计日志器"""
    global audit_logger
    if audit_logger is None:
        audit_logger = AuditLogger("./logs/audit")
    return audit_logger


# 使用示例
if __name__ == "__main__":
    logger = get_audit_logger()
    
    # 记录请求
    logger.log_request({"prompt": "创建材质"})
    
    # 开始操作
    logger.log_tool_execution("create_material", {"name": "MyMaterial"})
    
    # 模拟操作
    time.sleep(0.5)
    
    # 结束操作
    logger.end_tool_execution(True, result="Material_001")
