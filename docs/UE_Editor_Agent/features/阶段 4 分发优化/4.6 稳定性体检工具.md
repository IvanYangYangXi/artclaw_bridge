## 🔧 核心实现

```python
import os
import json
from typing import Dict, Any, List
import unreal

class HealthChecker:
    """健康检查器"""
    
    def __init__(self):
        self.checks: List[Dict[str, Any]] = []
    
    def run_all_checks(self) -> Dict[str, Any]:
        """运行所有检查"""
        self.checks = [
            self._check_plugin_installation(),
            self._check_mcp_connection(),
            self._check_python_dependencies(),
            self._check_skill_loading(),
            self._check_permissions()
        ]
        
        # 计算总分
        total_score = sum(check["score"] for check in self.checks)
        max_score = sum(check["max_score"] for check in self.checks)
        
        return {
            "overall_score": total_score,
            "max_score": max_score,
            "health_percentage": (total_score / max_score) * 100,
            "checks": self.checks,
            "recommendations": self._generate_recommendations()
        }
    
    def _check_plugin_installation(self) -> Dict[str, Any]:
        """检查插件安装"""
        plugin_path = "./Plugins/UEEditorAgent"
        exists = os.path.exists(plugin_path)
        
        return {
            "name": "插件安装",
            "status": "passed" if exists else "failed",
            "score": 10 if exists else 0,
            "max_score": 10,
            "details": f"插件路径: {plugin_path}",
            "recommendation": None if exists else "请重新安装插件"
        }
    
    def _check_mcp_connection(self) -> Dict[str, Any]:
        """检查MCP连接"""
        try:
            # 简化的连接测试
            connected = True  # 实际应测试WebSocket连接
            
            return {
                "name": "MCP连接",
                "status": "passed" if connected else "failed",
                "score": 15 if connected else 0,
                "max_score": 15,
                "details": "MCP服务器连接正常" if connected else "无法连接MCP服务器",
                "recommendation": None if connected else "检查MCP服务器配置"
            }
        except Exception as e:
            return {
                "name": "MCP连接",
                "status": "failed",
                "score": 0,
                "max_score": 15,
                "details": f"异常: {e}",
                "recommendation": "检查网络连接和MCP配置"
            }
    
    def _check_python_dependencies(self) -> Dict[str, Any]:
        """检查Python依赖"""
        required_packages = ["unreal", "requests"]
        missing = []
        
        for package in required_packages:
            try:
                __import__(package)
            except ImportError:
                missing.append(package)
        
        return {
            "name": "Python依赖",
            "status": "passed" if not missing else "warning",
            "score": 10 if not missing else 5,
            "max_score": 10,
            "details": f"缺失包: {missing}" if missing else "所有依赖已安装",
            "recommendation": f"pip install {' '.join(missing)}" if missing else None
        }
    
    def _check_skill_loading(self) -> Dict[str, Any]:
        """检查Skill加载"""
        skills_dir = "./skills"
        if not os.path.exists(skills_dir):
            return {
                "name": "Skill加载",
                "status": "warning",
                "score": 5,
                "max_score": 10,
                "details": "Skills目录不存在",
                "recommendation": "创建Skills目录并添加技能"
            }
        
        # 统计可加载的Skill
        skill_count = len([d for d in os.listdir(skills_dir) 
                          if os.path.isdir(os.path.join(skills_dir, d))])
        
        return {
            "name": "Skill加载",
            "status": "passed",
            "score": 10,
            "max_score": 10,
            "details": f"检测到 {skill_count} 个Skill",
            "recommendation": None
        }
    
    def _check_permissions(self) -> Dict[str, Any]:
        """检查权限"""
        # 检查文件写入权限
        test_file = "./test_write.tmp"
        try:
            with open(test_file, 'w') as f:
                f.write("test")
            os.remove(test_file)
            has_permission = True
        except:
            has_permission = False
        
        return {
            "name": "文件权限",
            "status": "passed" if has_permission else "failed",
            "score": 10 if has_permission else 0,
            "max_score": 10,
            "details": "有写入权限" if has_permission else "无写入权限",
            "recommendation": None if has_permission else "以管理员权限运行"
        }
    
    def _generate_recommendations(self) -> List[str]:
        """生成修复建议"""
        recommendations = []
        
        for check in self.checks:
            if check["recommendation"]:
                recommendations.append(f"{check['name']}: {check['recommendation']}")
        
        return recommendations
    
    def generate_report(self, report_path: str):
        """生成HTML报告"""
        result = self.run_all_checks()
        
        html = f"""
<!DOCTYPE html>
<html>
<head>
    <title>UE Editor Agent - 健康检查报告</title>
    <style>
        body {{ font-family: Arial; margin: 20px; }}
        .header {{ background: #0078D4; color: white; padding: 20px; }}
        .score {{ font-size: 48px; color: #107C10; }}
        .check {{ margin: 10px 0; padding: 10px; border-left: 4px solid; }}
        .passed {{ border-color: #107C10; background: #f0f8f0; }}
        .failed {{ border-color: #D13438; background: #fdf0f0; }}
        .warning {{ border-color: #FF8C00; background: #fff8f0; }}
    </style>
</head>
<body>
    <div class="header">
        <h1>UE Editor Agent 健康检查报告</h1>
        <div class="score">{result['health_percentage']:.0f}/100</div>
    </div>
    
    <h2>检查项详情</h2>
    """
        
        for check in result["checks"]:
            status_class = check["status"]
            html += f"""
    <div class="check {status_class}">
        <h3>{check['name']} - {check['score']}/{check['max_score']}</h3>
        <p>{check['details']}</p>
    </div>
    """
        
        if result["recommendations"]:
            html += """
    <h2>修复建议</h2>
    <ul>
    """
            for rec in result["recommendations"]:
                html += f"<li>{rec}</li>"
            
            html += """
    </ul>
    """
        
        html += """
</body>
</html>
"""
        
        os.makedirs(os.path.dirname(report_path), exist_ok=True)
        with open(report_path, 'w', encoding='utf-8') as f:
            f.write(html)
        
        print(f"[HealthCheck] 报告生成: {report_path}")


# 使用示例
if __name__ == "__main__":
    checker = HealthChecker()
    result = checker.run_all_checks()
    
    print(f"健康评分: {result['health_percentage']:.0f}/100")
    print(f"建议修复: {len(result['recommendations'])} 项")
    
    checker.generate_report("./reports/health_check.html")
