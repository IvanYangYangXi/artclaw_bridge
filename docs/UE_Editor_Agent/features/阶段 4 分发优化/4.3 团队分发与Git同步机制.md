import os
import subprocess
from typing import Dict, Any, List, Optional
import shutil

class GitSyncManager:
    """Git同步管理器"""
    
    def __init__(self, repo_url: str, local_dir: str):
        self.repo_url = repo_url
        self.local_dir = local_dir
        self.skills_subdir = "skills"
    
    def init_repository(self) -> bool:
        """初始化仓库"""
        try:
            if not os.path.exists(self.local_dir):
                # 克隆仓库
                subprocess.run([
                    "git", "clone", self.repo_url, self.local_dir
                ], check=True, capture_output=True)
            else:
                # 更新已有仓库
                subprocess.run(
                    ["git", "pull"],
                    cwd=self.local_dir,
                    check=True,
                    capture_output=True
                )
            
            print(f"[GitSync] 仓库初始化成功: {self.repo_url}")
            return True
            
        except subprocess.CalledProcessError as e:
            print(f"[GitSync] 仓库初始化失败: {e}")
            return False
    
    def sync_skills(self) -> Dict[str, Any]:
        """同步Skills"""
        try:
            # 拉取最新
            subprocess.run(
                ["git", "pull", "origin", "main"],
                cwd=self.local_dir,
                check=True,
                capture_output=True
            )
            
            skills_dir = os.path.join(self.local_dir, self.skills_subdir)
            if not os.path.exists(skills_dir):
                return {"success": True, "skills": []}
            
            # 扫描Skills
            skills = []
            for item in os.listdir(skills_dir):
                item_path = os.path.join(skills_dir, item)
                if os.path.isdir(item_path):
                    manifest_path = os.path.join(item_path, "manifest.json")
                    if os.path.exists(manifest_path):
                        skills.append(item)
            
            return {
                "success": True,
                "skills": skills,
                "count": len(skills)
            }
            
        except subprocess.CalledProcessError as e:
            return {
                "success": False,
                "error": str(e)
            }
    
    def publish_skill(self, skill_name: str, author: str, message: str) -> bool:
        """发布Skill到Git"""
        try:
            skills_dir = os.path.join(self.local_dir, self.skills_subdir)
            skill_dir = os.path.join(skills_dir, skill_name)
            
            if not os.path.exists(skill_dir):
                print(f"[GitSync] Skill不存在: {skill_name}")
                return False
            
            # 添加文件
            subprocess.run(
                ["git", "add", f"{self.skills_subdir}/{skill_name}"],
                cwd=self.local_dir,
                check=True,
                capture_output=True
            )
            
            # 提交
            commit_msg = f"Add skill: {skill_name}\nAuthor: {author}\n\n{message}"
            subprocess.run(
                ["git", "commit", "-m", commit_msg],
                cwd=self.local_dir,
                check=True,
                capture_output=True
            )
            
            # 推送
            subprocess.run(
                ["git", "push", "origin", "main"],
                cwd=self.local_dir,
                check=True,
                capture_output=True
            )
            
            print(f"[GitSync] Skill发布成功: {skill_name}")
            return True
            
        except subprocess.CalledProcessError as e:
            print(f"[GitSync] 发布失败: {e}")
            return False
    
    def get_sync_status(self) -> Dict[str, Any]:
        """获取同步状态"""
        try:
            # 检查是否有未提交更改
            status_result = subprocess.run(
                ["git", "status", "--porcelain"],
                cwd=self.local_dir,
                capture_output=True,
                text=True
            )
            
            # 获取最新提交
            log_result = subprocess.run(
                ["git", "log", "-1", "--format=%H|%an|%ae|%ad|%s"],
                cwd=self.local_dir,
                capture_output=True,
                text=True
            )
            
            has_changes = len(status_result.stdout.strip()) > 0
            
            sync_info = {
                "has_uncommitted_changes": has_changes,
                "latest_commit": None
            }
            
            if log_result.returncode == 0:
                parts = log_result.stdout.strip().split('|', 4)
                if len(parts) == 5:
                    sync_info["latest_commit"] = {
                        "hash": parts[0],
                        "author": parts[1],
                        "email": parts[2],
                        "date": parts[3],
                        "message": parts[4]
                    }
            
            return sync_info
            
        except Exception as e:
            return {"error": str(e)}


# 使用示例
if __name__ == "__main__":
    sync_mgr = GitSyncManager(
        repo_url="https://github.com/team/openclaw-skills.git",
        local_dir="./team_skills"
    )
    
    # 初始化
    sync_mgr.init_repository()
    
    # 同步
    result = sync_mgr.sync_skills()
    print(f"同步结果: {result}")
