### Skill打包器

```python
import os
import json
import zipfile
from typing import Dict, Any
import shutil

class SkillPackager:
    """技能包打包器"""
    
    def __init__(self, skills_dir: str):
        self.skills_dir = skills_dir
    
    def package_skill(self, skill_name: str, output_dir: str) -> str:
        """打包技能为.openclaw文件"""
        skill_dir = os.path.join(self.skills_dir, skill_name)
        
        if not os.path.exists(skill_dir):
            raise FileNotFoundError(f"Skill目录不存在: {skill_dir}")
        
        # 读取清单
        manifest_path = os.path.join(skill_dir, "manifest.json")
        with open(manifest_path, 'r', encoding='utf-8') as f:
            manifest = json.load(f)
        
        # 创建输出文件
        package_name = f"{skill_name}_v{manifest['version']}.openclaw"
        output_path = os.path.join(output_dir, package_name)
        
        # 打包
        with zipfile.ZipFile(output_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
            for root, _, files in os.walk(skill_dir):
                for file in files:
                    file_path = os.path.join(root, file)
                    arcname = os.path.relpath(file_path, skill_dir)
                    zipf.write(file_path, arcname)
        
        print(f"[Packager] 技能包创建成功: {output_path}")
        return output_path
    
    def install_skill(self, package_path: str, install_dir: str) -> bool:
        """安装技能包"""
        try:
            # 解压
            skill_name = os.path.basename(package_path).split('_v')[0]
            extract_dir = os.path.join(install_dir, skill_name)
            
            with zipfile.ZipFile(package_path, 'r') as zipf:
                zipf.extractall(extract_dir)
            
            # 安装依赖
            self._install_dependencies(extract_dir)
            
            print(f"[Packager] 技能安装成功: {skill_name}")
            return True
            
        except Exception as e:
            print(f"[Packager] 安装失败: {e}")
            return False
    
    def _install_dependencies(self, skill_dir: str):
        """安装Python依赖"""
        req_path = os.path.join(skill_dir, "requirements.txt")
