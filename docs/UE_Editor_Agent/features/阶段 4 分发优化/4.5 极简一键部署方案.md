## 🔧 核心实现

```python
import os
import shutil
import zipfile
from typing import Dict, Any, List
import subprocess

class Deployer:
    """部署器"""
    
    def __init__(self, plugin_source: str, skills_dir: str):
        self.plugin_source = plugin_source
        self.skills_dir = skills_dir
    
    def create_package(self, output_path: str) -> str:
        """创建部署包"""
        # 创建临时目录
        temp_dir = "./temp_deploy"
        os.makedirs(temp_dir, exist_ok=True)
        
        try:
            # 复制插件
            plugin_dest = os.path.join(temp_dir, "Plugins", "UEEditorAgent")
            shutil.copytree(self.plugin_source, plugin_dest)
            
            # 复制Skills
            skills_dest = os.path.join(temp_dir, "Skills")
            shutil.copytree(self.skills_dir, skills_dest)
            
            # 创建安装脚本
            self._create_install_script(temp_dir)
            
            # 打包
            package_path = self._create_zip(temp_dir, output_path)
            
            print(f"[Deployer] 部署包创建成功: {package_path}")
            return package_path
            
        finally:
            # 清理临时文件
            shutil.rmtree(temp_dir, ignore_errors=True)
    
    def _create_install_script(self, temp_dir: str):
        """创建安装脚本"""
        script_content = '''
import os
import sys
import shutil

def install():
    # 检测UE项目
    project_dir = os.getcwd()
    uproject_files = [f for f in os.listdir(project_dir) if f.endswith('.uproject')]
    
    if not uproject_files:
        print("错误：当前目录不是UE项目")
        return False
    
    # 安装插件
    plugin_src = os.path.join(os.path.dirname(__file__), "Plugins", "UEEditorAgent")
    plugin_dst = os.path.join(project_dir, "Plugins", "UEEditorAgent")
    
    if os.path.exists(plugin_dst):
        shutil.rmtree(plugin_dst)
    
    shutil.copytree(plugin_src, plugin_dst)
    
    # 安装Skills
    skills_src = os.path.join(os.path.dirname(__file__), "Skills")
    skills_dst = os.path.join(os.path.expanduser("~"), ".openclaw", "skills")
    
    os.makedirs(skills_dst, exist_ok=True)
    shutil.copytree(skills_src, skills_dst, dirs_exist_ok=True)
    
    print("安装成功！请重启UE编辑器。")
    return True

if __name__ == "__main__":
    install()
'''
        
        script_path = os.path.join(temp_dir, "install.py")
        with open(script_path, 'w', encoding='utf-8') as f:
            f.write(script_content)
    
    def _create_zip(self, source_dir: str, output_path: str) -> str:
        """创建ZIP包"""
        zip_path = os.path.join(output_path, "UEAgent_Deploy.zip")
        
        with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
            for root, _, files in os.walk(source_dir):
                for file in files:
                    file_path = os.path.join(root, file)
                    arcname = os.path.relpath(file_path, source_dir)
                    zipf.write(file_path, arcname)
        
        return zip_path
    
    def install_to_project(self, package_path: str, project_dir: str) -> bool:
        """安装到UE项目"""
        try:
            # 解压
            with zipfile.ZipFile(package_path, 'r') as zipf:
                zipf.extractall(project_dir)
            
            # 运行安装脚本
            install_script = os.path.join(project_dir, "install.py")
            if os.path.exists(install_script):
                subprocess.run(
                    [sys.executable, install_script],
                    cwd=project_dir,
                    check=True
                )
                
                # 清理安装脚本
                os.remove(install_script)
            
            print(f"[Deployer] 安装成功: {project_dir}")
            return True
            
        except Exception as e:
            print(f"[Deployer] 安装失败: {e}")
            return False


# 使用示例
if __name__ == "__main__":
    deployer = Deployer(
        plugin_source="./src/UEEditorAgent",
        skills_dir="./skills"
    )
    
    # 创建部署包
    package = deployer.create_package("./deployments")
    
    # 安装到项目
    deployer.install_to_project(package, "D:/MyProject/MyGame")
