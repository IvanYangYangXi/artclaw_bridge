## 🔧 核心实现

```python
import unreal
from enum import Enum
from typing import Dict, List, Any

class EditorMode(Enum):
    LEVEL_EDITOR = "level"
    BLUEPRINT = "blueprint"
    MATERIAL = "material"
    SEQUENCER = "sequencer"
    UNKNOWN = "unknown"

class ModeFilter:
    """编辑器模式过滤器"""
    
    def __init__(self):
        self.mode_tools: Dict[EditorMode, List[str]] = {
            EditorMode.LEVEL_EDITOR: [
                "spawn_actor",
                "delete_actor",
                "move_actor",
                "batch_rename"
            ],
            EditorMode.BLUEPRINT: [
                "create_node",
                "connect_pins",
                "compile_blueprint"
            ],
            EditorMode.MATERIAL: [
                "create_node",
                "connect_material_pins"
            ],
            EditorMode.SEQUENCER: [
                "add_track",
                "set_keyframe"
            ]
        }
    
    def get_current_mode(self) -> EditorMode:
        """获取当前编辑器模式"""
        # 简化的模式检测
        if unreal.EditorLevelLibrary.get_selected_level_actors():
            return EditorMode.LEVEL_EDITOR
        elif unreal.EditorUtilityLibrary.get_selected_blueprint():
            return EditorMode.BLUEPRINT
        return EditorMode.UNKNOWN
    
    def filter_tools(self, tools: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """根据当前模式过滤Tools"""
        current_mode = self.get_current_mode()
        allowed_tools = self.mode_tools.get(current_mode, [])
        
        filtered = []
        for tool in tools:
            if tool.get("name") in allowed_tools:
                filtered.append(tool)
        
        return filtered


# 使用示例
if __name__ == "__main__":
    filter = ModeFilter()
    current_mode = filter.get_current_mode()
    print(f"当前模式: {current_mode.value}")
    
    # 示例Tools列表
    tools = [
        {"name": "spawn_actor", "description": "生成Actor"},
        {"name": "create_node", "description": "创建蓝图节点"}
    ]
    
    filtered = filter.filter_tools(tools)
    print(f"过滤后的Tools: {[t['name'] for t in filtered]}")
