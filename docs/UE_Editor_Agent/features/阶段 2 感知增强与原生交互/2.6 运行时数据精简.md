## 🔧 核心实现

```python
import json
from typing import Dict, Any, List

class DataCompressor:
    """数据压缩器"""
    
    def __init__(self):
        self.compression_rules = {
            # 移除不必要的字段
            "remove_fields": ["__uuid", "__interpreter", "__type"],
            # 简化路径
            "simplify_paths": True,
            # 限制数组长度
            "max_array_length": 10
        }
    
    def compress(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """压缩数据"""
        if isinstance(data, dict):
            return self._compress_dict(data)
        elif isinstance(data, list):
            return self._compress_list(data)
        else:
            return data
    
    def _compress_dict(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """压缩字典"""
        compressed = {}
        
        for key, value in data.items():
            # 跳过不需要的字段
            if key in self.compression_rules["remove_fields"]:
                continue
            
            # 递归压缩值
            compressed[key] = self.compress(value)
        
        return compressed
    
    def _compress_list(self, data: List[Any]) -> List[Any]:
        """压缩列表"""
        if not data:
            return data
        
        max_len = self.compression_rules["max_array_length"]
        if len(data) > max_len:
            # 保留前N个，添加省略标记
            return data[:max_len] + [f"... ({len(data) - max_len} more)"]
        
        return [self.compress(item) for item in data]
    
    def estimate_tokens(self, data: Dict[str, Any]) -> int:
        """估算Token数量"""
        text = json.dumps(data, ensure_ascii=False)
        # 粗略估算：每个token约4个字符
        return len(text) // 4


# 使用示例
if __name__ == "__main__":
    compressor = DataCompressor()
    
    # 示例数据
    data = {
        "actors": [{"name": f"Actor_{i}", "path": f"/Game/Actor_{i}"} for i in range(100)],
        "__uuid": "12345",
        "__type": "context"
    }
    
    # 压缩前
    before_tokens = compressor.estimate_tokens(data)
    print(f"压缩前Token: {before_tokens}")
    
    # 压缩
    compressed = compressor.compress(data)
    after_tokens = compressor.estimate_tokens(compressed)
    print(f"压缩后Token: {after_tokens}")
    print(f"压缩率: {(1 - after_tokens/before_tokens) * 100:.1f}%")
