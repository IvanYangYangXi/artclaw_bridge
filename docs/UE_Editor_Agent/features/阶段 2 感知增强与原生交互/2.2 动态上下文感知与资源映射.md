## 🎯 目标与里程碑

### 核心目标
构建**动态上下文感知系统**，实时映射UE资源（Actor、资产、关卡）为MCP Resources，让AI能理解和操作当前编辑器状态。

### 验收标准
- [ ] 自动发现当前关卡所有Actor
- [ ] 映射选中Actor为MCP Resource
- [ ] 资源变更实时同步（延迟<100ms）
- [ ] 支持查询资源属性（位置、旋转、缩放）
- [ ] 支持资源层级关系（父子Actor）

## 🔧 核心实现

```python
import unreal
from typing import Dict, List, Any, Optional
import json
import time

class ResourceMapper:
    """资源映射器 - 将UE对象映射为MCP Resources"""
    
    def __init__(self):
        self.resource_cache: Dict[str, Any] = {}
        self.cache_timestamp = 0
        self.cache_ttl = 0.1  # 100ms缓存
    
    def get_level_resources(self) -> List[Dict[str, Any]]:
        """获取当前关卡所有Actor作为资源"""
        current_time = time.time()
        if current_time - self.cache_timestamp < self.cache_ttl:
            return self.resource_cache.get("level_actors", [])
        
        actors = unreal.EditorLevelLibrary.get_all_level_actors()
        resources = []
        
        for actor in actors:
            resource = {
                "uri": f"actor://{actor.get_path_name()}",
                "name": actor.get_name(),
                "type": "actor",
                "mimeType": "application/unreal-actor",
                "metadata": {
                    "class": actor.get_class().get_name(),
                    "location": str(actor.get_actor_location()),
                    "rotation": str(actor.get_actor_rotation()),
                    "scale": str(actor.get_actor_scale3d()),
                    "tags": actor.get_tags()
                }
            }
            resources.append(resource)
        
        self.resource_cache["level_actors"] = resources
        self.cache_timestamp = current_time
        
        return resources
    
    def get_selected_resources(self) -> List[Dict[str, Any]]:
        """获取选中资源"""
        selected_actors = unreal.EditorLevelLibrary.get_selected_level_actors()
        selected_assets = unreal.EditorUtilityLibrary.get_selected_assets()
        
        resources = []
        
        # 添加Actor
        for actor in selected_actors:
            resources.append({
                "uri": f"actor://{actor.get_path_name()}",
                "name": actor.get_name(),
                "type": "actor",
                "metadata": {
                    "class": actor.get_class().get_name()
                }
            })
        
        # 添加资产
        for asset in selected_assets:
            resources.append({
                "uri": f"asset://{asset.get_path_name()}",
                "name": asset.get_name(),
                "type": "asset",
                "metadata": {
                    "class": asset.get_class().get_name()
                }
            })
        
        return resources
    
    def read_resource(self, uri: str) -> str:
        """读取资源内容"""
        if uri.startswith("actor://"):
            return self._read_actor(uri.replace("actor://", ""))
        elif uri.startswith("asset://"):
            return self._read_asset(uri.replace("asset://", ""))
        else:
            raise ValueError(f"不支持的资源URI: {uri}")
    
    def _read_actor(self, path: str) -> str:
        """读取Actor信息"""
        actor = unreal.load_object(None, path)
        if not actor:
            raise ValueError(f"Actor未找到: {path}")
        
        info = {
            "name": actor.get_name(),
            "class": actor.get_class().get_name(),
            "location": str(actor.get_actor_location()),
            "rotation": str(actor.get_actor_rotation()),
            "scale": str(actor.get_actor_scale3d()),
            "components": [comp.get_name() for comp in actor.get_components()],
            "tags": actor.get_tags()
        }
        
        return json.dumps(info, indent=2, ensure_ascii=False)
    
    def _read_asset(self, path: str) -> str:
        """读取资产信息"""
        asset = unreal.load_object(None, path)
        if not asset:
            raise ValueError(f"资产未找到: {path}")
        
        info = {
            "name": asset.get_name(),
            "class": asset.get_class().get_name(),
            "path": asset.get_path_name()
        }
        
        return json.dumps(info, indent=2, ensure_ascii=False)


# MCP Resources接口
class McpResources:
    """MCP Resources实现"""
    
    def __init__(self):
        self.mapper = ResourceMapper()
    
    def list_resources(self) -> List[Dict[str, Any]]:
        """列出所有可用资源"""
        return self.mapper.get_level_resources()
    
    def read_resource(self, uri: str) -> str:
        """读取资源内容"""
        return self.mapper.read_resource(uri)
    
    def subscribe_resource_changes(self, callback: callable):
        """订阅资源变更"""
        # 简化的变更订阅
        def check_changes():
            old_resources = self.mapper.resource_cache.get("level_actors", [])
            new_resources = self.mapper.get_level_resources()
            
            if len(old_resources) != len(new_resources):
                callback(new_resources)
        
        # 定时检查（实际应使用UE的事件系统）
        import threading
        def loop():
            while True:
                check_changes()
                time.sleep(0.1)
        
        threading.Thread(target=loop, daemon=True).start()
