import unreal
import traceback
from typing import Dict, Any, List, Optional, Callable
import re

class ErrorAnalyzer:
    """错误分析器"""
    
    def __init__(self):
        self.error_patterns = {
            r'.*does not exist': {
                "type": "not_exist",
                "suggestion": "检查对象路径是否正确"
            },
            r'.*already exists': {
                "type": "already_exists",
                "suggestion": "删除已存在的对象或重命名"
            },
            r'.*permission denied': {
                "type": "permission_denied",
                "suggestion": "检查文件权限或以管理员运行"
            },
            r'.*out of memory': {
                "type": "out_of_memory",
                "suggestion": "释放内存或重启编辑器"
            }
        }
    
    def analyze(self, error: Exception) -> Dict[str, Any]:
        """分析错误"""
        error_str = str(error).lower()
        
        for pattern, info in self.error_patterns.items():
            if re.match(pattern, error_str, re.IGNORECASE):
                return {
                    "type": info["type"],
                    "message": str(error),
                    "suggestion": info["suggestion"],
                    "auto_fixable": info["type"] in ["already_exists"]
                }
        
        return {
            "type": "unknown",
            "message": str(error),
            "suggestion": "查看日志获取详细信息",
            "auto_fixable": False
        }


class SelfHealingLoop:
    """自修复循环"""
    
    def __init__(self):
        self.analyzer = ErrorAnalyzer()
        self.max_retries = 3
    
    def execute_with_retry(self, operation: Callable, 
                          context: Dict[str, Any]) -> Dict[str, Any]:
        """
        执行操作并自动重试
        
        Returns:
            {
                "success": bool,
                "result": Any,
                "error": Optional[Dict],
                "retries": int
            }
        """
        last_error = None
        
        for attempt in range(self.max_retries):
            try:
                result = operation()
                return {
                    "success": True,
                    "result": result,
                    "error": None,
                    "retries": attempt
                }
            except Exception as e:
                error_info = self.analyzer.analyze(e)
                last_error = error_info
                
                # 尝试自动修复
                if error_info["auto_fixable"] and attempt < self.max_retries - 1:
                    fix_success = self._try_auto_fix(error_info, context)
                    if fix_success:
                        unreal.log(f"[SelfHeal] 自动修复成功: {error_info['type']}")
                        continue
                
                # 记录错误日志
                unreal.log_error(f"[SelfHeal] 操作失败: {e}\n{traceback.format_exc()}")
                
                if attempt == self.max_retries - 1:
                    break
                
                # 等待后重试
                time.sleep(0.5 * (attempt + 1))
        
        return {
            "success": False,
            "result": None,
            "error": last_error,
            "retries": self.max_retries
        }
    
    def _try_auto_fix(self, error_info: Dict[str, Any], 
                     context: Dict[str, Any]) -> bool:
        """尝试自动修复"""
        error_type = error_info["type"]
        
        if error_type == "already_exists":
            # 自动重命名
            if "path" in context:
                old_path = context["path"]
                new_path = old_path + "_fixed"
                context["path"] = new_path
                unreal.log(f"[AutoFix] 自动重命名: {old_path} -> {new_path}")
                return True
        
        return False
