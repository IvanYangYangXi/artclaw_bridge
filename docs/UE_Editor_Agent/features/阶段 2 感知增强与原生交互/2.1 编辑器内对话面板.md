### C++ Slate集成

```cpp
// Source/UEEditorAgent/Private/SUEAgentChatTab.cpp

#include "SUEAgentChatTab.h"
#include "Widgets/Input/SMultiLineEditableTextBox.h"
#include "Widgets/Text/STextBlock.h"
#include "Widgets/Layout/SBox.h"
#include "Widgets/Layout/SScrollBox.h"
#include "WebBrowserModule.h"
#include "IWebBrowserCookieManager.h"
#include "IWebBrowserSingleton.h"

#define LOCTEXT_NAMESPACE "SUEAgentChatTab"

void SUEAgentChatTab::Construct(const FArguments& InArgs)
{
    // 创建WebBrowser
    TSharedPtr<IWebBrowserSingleton> WebBrowserSingleton = IWebBrowserModule::Get().GetSingleton();
    WebBrowser = WebBrowserSingleton->CreateBrowser();
    
    // 加载聊天面板HTML
    FString HtmlPath = FPaths::ProjectPluginsDir() / TEXT("UEEditorAgent/Content/Python/chat_panel.html");
    WebBrowser->LoadURL(FString::Printf(TEXT("file:///%s"), *HtmlPath));
    
    // 绑定消息处理
    WebBrowser->OnBeforePopup().AddLambda([](FString URL, FString Frame) -> bool {
        return true; // 阻止弹窗
    });
    
    ChildSlot
    [
        SNew(SVerticalBox)
        + SVerticalBox::Slot()
        .FillHeight(1.0f)
        [
            WebBrowser->GetWidget()
        ]
    ];
}

void SUEAgentChatTab::SendMessage(const FString& Message)
{
    // 通过JavaScript发送消息到网页
    FString JS = FString::Printf(TEXT("addMessage('%s', true);"), *Message);
    WebBrowser->ExecuteJavascript(JS);
}

void SUEAgentChatTab::ReceiveMessage(const FString& Message)
{
    // 接收AI响应并显示
    FString JS = FString::Printf(TEXT("addMessage('%s', false);"), *Message);
    WebBrowser->ExecuteJavascript(JS);
}

#undef LOCTEXT_NAMESPACE
```

        """处理用户消息"""
        # 添加到历史

        })
