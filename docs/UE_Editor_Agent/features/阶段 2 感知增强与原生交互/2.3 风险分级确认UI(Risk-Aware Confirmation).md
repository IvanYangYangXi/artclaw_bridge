# 2.3 风险分级确认UI(Risk-Aware Confirmation)

**阶段**：阶段2 - 感知增强与原生交互  
**重要性**：⭐⭐⭐⭐⭐ (P0)  
**依赖**：2.2 动态上下文感知  
**预估工时**：2-3天

## 🎯 目标与里程碑

### 核心目标
构建**风险分级确认UI**，根据AI操作的风险等级（低/中/高/致命）显示不同样式的确认对话框，提供**撤销/重试/跳过**选项。

### 验收标准
- [ ] 风险等级自动计算（基于影响范围）
- [ ] 4种UI样式（对应4个风险等级）
- [ ] 支持预览模式（不实际执行）
- [ ] 支持"信任此AI"一键授权
- [ ] 确认对话框响应时间<100ms

## 🔧 核心实现

```python
import unreal
from enum import Enum
from typing import Dict, Any, List, Callable, Optional

class RiskLevel(Enum):
    LOW = "low"          # 绿色 - 安全操作
    MEDIUM = "medium"    # 黄色 - 需谨慎
    HIGH = "high"        # 橙色 - 高风险
    CRITICAL = "critical" # 红色 - 致命风险

class RiskCalculator:
    """风险计算器"""
    
    def calculate_risk(self, operation: Dict[str, Any]) -> RiskLevel:
        """计算操作风险等级"""
        op_type = operation.get("type", "")
        params = operation.get("params", {})
        
        # CRITICAL风险：删除、格式化
        if op_type in ["delete_actor", "delete_asset", "format_disk"]:
            return RiskLevel.CRITICAL
        
        # HIGH风险：批量修改、文件覆盖
        if op_type == "batch_modify":
            count = params.get("count", 0)
            if count > 10:
                return RiskLevel.HIGH
            return RiskLevel.MEDIUM
        
        # MEDIUM风险：修改、写入
        if op_type in ["modify_actor", "write_file"]:
            return RiskLevel.MEDIUM
        
        # LOW风险：读取、查询
        if op_type in ["read_actor", "query_asset", "select_actor"]:
            return RiskLevel.LOW
        
        return RiskLevel.LOW


class RiskAwareUI:
    """风险感知UI"""
    
    def __init__(self):
        self.calculator = RiskCalculator()
        self.trusted_ai = set()  # 已信任的AI会话
    
    def confirm_operation(self, operation: Dict[str, Any], 
                         ai_session_id: str) -> bool:
        """
        确认操作
        
        Returns:
            True: 用户确认执行
            False: 用户取消
        """
        # 检查是否已信任此AI
        if ai_session_id in self.trusted_ai:
            unreal.log(f"[RiskUI] AI {ai_session_id} 已信任，跳过确认")
            return True
        
        risk_level = self.calculator.calculate_risk(operation)
        
        # 低风险直接通过
        if risk_level == RiskLevel.LOW:
            return True
        
        # 显示确认对话框
        return self._show_confirmation_dialog(operation, risk_level, ai_session_id)
    
    def _show_confirmation_dialog(self, operation: Dict[str, Any],
                                  risk_level: RiskLevel,
                                  ai_session_id: str) -> bool:
        """显示确认对话框"""
        op_name = operation.get("name", "未知操作")
        op_details = operation.get("details", "")
        
        # 构建对话框文本
        title = f"AI操作确认 - {risk_level.value.upper()}"
        message = f"操作: {op_name}\n\n{op_details}\n\n是否执行此操作？"
        
        # 根据风险等级选择样式
        if risk_level == RiskLevel.CRITICAL:
            return self._show_critical_dialog(title, message, ai_session_id)
        elif risk_level == RiskLevel.HIGH:
            return self._show_high_risk_dialog(title, message, ai_session_id)
        elif risk_level == RiskLevel.MEDIUM:
            return self._show_medium_risk_dialog(title, message, ai_session_id)
        else:
            return True
    
    def _show_critical_dialog(self, title: str, message: str,
                             ai_session_id: str) -> bool:
        """致命风险对话框（红色）"""
        # 使用UE的对话框系统
        dialog = unreal.EditorDialog(
            title=title,
            message=message,
            message_type=unreal.AppMsgType.YES_NO_CANCEL
        )
        
        # 添加自定义按钮
        result = dialog.show_custom(
            buttons=[
                ("取消", unreal.AppReturnType.CANCEL),
                ("预览", unreal.AppReturnType.NO),
                ("信任此AI", unreal.AppReturnType.YES)
            ],
            default_button=0,
            cancel_button=0
        )
        
        if result == unreal.AppReturnType.YES:
            # 信任此AI
            self.trusted_ai.add(ai_session_id)
            return True
        elif result == unreal.AppReturnType.NO:
            # 预览模式
            return self._preview_operation()
        else:
            # 取消
            return False
    
    def _show_high_risk_dialog(self, title: str, message: str,
                              ai_session_id: str) -> bool:
        """高风险对话框（橙色）"""
        dialog = unreal.EditorDialog(title=title, message=message)
        result = dialog.show_custom(
            buttons=[
                ("取消", unreal.AppReturnType.CANCEL),
                ("执行", unreal.AppReturnType.OK)
            ],
            default_button=0
        )
        return result == unreal.AppReturnType.OK
    
    def _show_medium_risk_dialog(self, title: str, message: str,
                                ai_session_id: str) -> bool:
        """中风险对话框（黄色）"""
        dialog = unreal.EditorDialog(title=title, message=message)
        result = dialog.show_custom(
            buttons=[
                ("否", unreal.AppReturnType.NO),
                ("是", unreal.AppReturnType.YES)
            ],
            default_button=1
        )
        return result == unreal.AppReturnType.YES
    
    def _preview_operation(self) -> bool:
        """预览模式（不实际执行）"""
        # 简化的预览实现
        unreal.log_warning("[RiskUI] 预览模式：操作未执行")
        unreal.EditorDialog(
            title="预览模式",
            message="操作预览完成（实际未执行）"
        ).show()
        return False


# 使用示例
if __name__ == "__main__":
    ui = RiskAwareUI()
    
    # 示例操作
    operation = {
        "type": "delete_actor",
        "name": "删除选中Actor",
        "details": "将永久删除以下Actor:\n- StaticMeshActor_1\n- StaticMeshActor_2\n\n此操作不可撤销！"
    }
    
    # 显示确认对话框
    confirmed = ui.confirm_operation(operation, ai_session_id="ai_001")
    
    if confirmed:
        unreal.log("用户确认执行操作")
    else:
        unreal.log("用户取消操作")
